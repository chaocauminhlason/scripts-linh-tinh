local cachedPlayerId = nil

local function getPlayerId()
    if cachedPlayerId then 
        return cachedPlayerId 
    end
    
    local petsFolder = Workspace:FindFirstChild("Pets")
    if not petsFolder then 
        return nil 
    end
    
    for _, pet in pairs(petsFolder:GetChildren()) do
        if pet:IsA("BasePart") then
            local playerId = pet:GetAttribute("PlayerId")
            if playerId then
                cachedPlayerId = playerId
                return playerId
            end
        end
    end
    
    return nil
end

local function hasDeadPet()
    local petsFolder = Workspace:FindFirstChild("Pets")
    if not petsFolder then 
        return false 
    end
    
    local playerId = getPlayerId()
    if not playerId then 
        return false 
    end
    
    local deadCount = 0
    local totalCount = 0
    
    for _, pet in pairs(petsFolder:GetChildren()) do
        if pet:GetAttribute("PlayerId") == playerId then
            totalCount = totalCount + 1
            local healthPart = pet:FindFirstChild("Health")
            if healthPart then
                local health = tonumber(healthPart.Value) or 0
                if health <= 0 then 
                    deadCount = deadCount + 1 
                end
            end
        end
    end
    
    return deadCount > 0 and totalCount > 0
end

local function getRecoverPosition()
    local area = Workspace:FindFirstChild("Area")
    if not area then 
        return nil 
    end
    
    for _, zone in pairs(area:GetChildren()) do
        local serverZone = zone:FindFirstChild("ServerZone")
        if serverZone then
            local recover = serverZone:FindFirstChild("Recover")
            if recover then
                if recover:IsA("Model") then
                    local root = recover.PrimaryPart or recover:FindFirstChildOfClass("Part")
                    return root and root.Position
                elseif recover:IsA("BasePart") then
                    return recover.Position
                end
            end
        end
    end
    
    return nil
end

local function findUIElementByName(parent, name)
    if not parent then return nil end
    for _, child in ipairs(parent:GetDescendants()) do
        if child.Name == name then 
            return child 
        end
    end
    return nil
end

local function ClickButtonExact(button, debugName)
    if not button then return end
    
    local absPos = button.AbsolutePosition
    local absSize = button.AbsoluteSize
    
    local centerX = absPos.X + (absSize.X / 2)
    local finalY = absPos.Y + (absSize.Y / 2)
    
    local screenGui = button:FindFirstAncestorOfClass("ScreenGui")
    if screenGui and not screenGui.IgnoreGuiInset then 
        finalY = finalY + GuiService:GetGuiInset().Y 
    end
    
    VirtualInputManager:SendMouseButtonEvent(centerX, finalY, 0, true, game, 1)
    task.wait(0.1)
    VirtualInputManager:SendMouseButtonEvent(centerX, finalY, 0, false, game, 1)
end

local function SafeExitRoutine()
    local exitBtn = findUIElementByName(playerGui, "BgLeave")
    
    if exitBtn and exitBtn.AbsoluteSize.X > 0 then 
        ClickButtonExact(exitBtn, "Nút Thoát") 
    else 
        autoExitTriggered = false 
        return 
    end
    
    task.wait(1.5)
    
    local mainGui = playerGui:FindFirstChild("MainGui")
    if mainGui then
        local screenGui = mainGui:FindFirstChild("ScreenGui")
        if screenGui and screenGui:FindFirstChild("ConfirmView") and screenGui.ConfirmView.Visible then
            local fmBottom = screenGui.ConfirmView:FindFirstChild("FmBottom")
            if fmBottom and fmBottom:FindFirstChild("BtOk") and fmBottom.BtOk.AbsoluteSize.X > 0 then
                for i = 1, 3 do 
                    ClickButtonExact(fmBottom.BtOk, "Nút Yes") 
                    task.wait(0.2) 
                end
            else 
                autoExitTriggered = false 
                return 
            end
        else 
            autoExitTriggered = false 
            return 
        end
    end
    
    task.wait(2.5)
    
    local checkLabStage = findUIElementByName(playerGui, "LabStage")
    local currentStageText = checkLabStage and checkLabStage.Text or "Unknown"
    
    if currentStageText == "Unknown" or not string.match(currentStageText, "20/20") then 
        autoExitTriggered = false
        stage20ClockStarted = false 
    else 
        autoExitTriggered = false 
    end
end

-- Hàm xử lý tìm Boss
local function IsBoss(model)
    if not model:IsA("Model") then return false end
    local hum = model:FindFirstChildOfClass("Humanoid")
    local hrp = model:FindFirstChild("HumanoidRootPart")
    if not hum or not hrp then return false end
    if hum.MaxHealth and hum.MaxHealth >= 100000 then
        return true
    end
    return false
end

local function FindBossByName(keywords)
    for _,v in pairs(workspace:GetDescendants()) do
        if v:IsA("Model") and IsBoss(v) then
            local n = v.Name:lower()
            for _,k in pairs(keywords) do
                if n:find(k) then
                    return v:FindFirstChild("HumanoidRootPart")
                end
            end
        end
    end
    return nil
end

    end
end

-- ====================================================================
-- CAM: Auto Catch (Phiên bản Chuẩn Giao Thức Máy Chủ - Tích hợp Complete)
-- ====================================================================
task.spawn(function()
    while task.wait(0.5) do
        if camSettings.autoCatch and not camSettings.isHealing then
            local char = LocalPlayer.Character
            local hrp = char and char:FindFirstChild("HumanoidRootPart")
            
            if hrp then
                local playerId = getPlayerId()
                local monstersFolder = Workspace:FindFirstChild("Monsters")
                
                if monstersFolder then
                    local targetCatch = nil
                    local closestDist = math.huge
                    
                    -- BƯỚC 1: TÌM QUÁI (Đã bao gồm bộ lọc VIP nếu bạn đang dùng)
                    for _, monster in pairs(monstersFolder:GetChildren()) do
                        if monster:IsA("BasePart") or monster:IsA("Model") then
                            local id = tonumber(monster.Name:match("Monster_(%d+)"))
                            if id then
                                local catchTaken = monster:GetAttribute("CatchTakenPlayerId")
                                -- Kiểm tra xem quái có bị người khác lấy chưa
                                if not catchTaken or catchTaken == playerId then
                                    local pos = monster:IsA("Model") and monster.PrimaryPart and monster.PrimaryPart.Position or monster.Position
                                    local dist = (pos - hrp.Position).Magnitude
                                    
                                    if dist < closestDist then 
                                        closestDist = dist
                                        targetCatch = monster 
                                    end
                                end
                            end
                        end
                    end
                    
                    -- BƯỚC 2: TIẾN HÀNH BẮT CHUẨN GIAO THỨC
                    if targetCatch and MonsterSystem then
                        local targetId = tonumber(targetCatch.Name:match("Monster_(%d+)"))
                        local targetPos = targetCatch:IsA("Model") and targetCatch.PrimaryPart and targetCatch.PrimaryPart.Position or targetCatch.Position
                        
                        -- 2.1 Bay tới trên đầu quái
                        hrp.CFrame = CFrame.new(targetPos + Vector3.new(0, 3, 0))
                        
                        -- 2.2 Gửi tín hiệu BẮT ĐẦU (Start)
                        local success, isStarted, errData = pcall(MonsterSystem.ClientCatchMonsterStart, targetId)
                        
                        -- Nếu túi đồ đầy, game sẽ trả về isStarted = false
                        if success and isStarted == false then
                            print("⚠️ TÚI ĐỒ CỦA BẠN ĐÃ ĐẦY! Hãy dọn dẹp hành trang.")
                            -- Tạm dừng Auto Catch 5 giây để tránh spam server
                            task.wait(5) 
                        elseif success then
                            -- Chờ 1.5 giây để game chạy hiệu ứng ném bóng
                            task.wait(1.5)
                            
                            -- 2.3 Gửi tín hiệu KẾT THÚC (Complete) ĐỂ CHỐT ĐƠN!
                            -- Đây là bước quan trọng nhất mà bản cũ còn thiếu
                            pcall(MonsterSystem.ClientCatchMonsterComplete, targetId)
                        end
                    end
                end
            end
        end
    end
end)

task.spawn(function()
    while task.wait(1) do
        if camSettings.autoHeal and not camSettings.isHealing then
            pcall(function()
                local petsFolder = Workspace:FindFirstChild("Pets")
                if not petsFolder then return end
                
                local myId = tostring(LocalPlayer.UserId)
                local hasDead = false
                
                -- BƯỚC 1: KIỂM TRA TÌNH TRẠNG PET
                for _, pet in pairs(petsFolder:GetChildren()) do
                    if string.find(pet.Name, myId) then
                        local hpPart = pet:FindFirstChild("Health")
                        if hpPart then
                            local hpStr = tostring(hpPart.Value)
                            local currentHp = tonumber(string.split(hpStr, ",")[1])
                            
                            if currentHp and currentHp <= 0 then
                                hasDead = true
                                break
                            end
                        end
                    end
                end

                -- BƯỚC 2: TIẾN HÀNH HỒI MÁU NẾU PET CHẾT
                if hasDead then
                    camSettings.isHealing = true
                    print("🚑 [Auto Heal] Phát hiện Pet chết! Bắt đầu quy trình hồi máu...")
                    
                    local char = LocalPlayer.Character
                    local hrp = char and char:FindFirstChild("HumanoidRootPart")
                    
                    if hrp then
                        -- Lưu lại bãi Farm
                        camSettings.savedPosition = hrp.CFrame
                        
                        -- Dò trạm Recover trên map hiện tại
                        local bestRecoverPos = nil
                        local minDist = math.huge
                        local areaFolder = Workspace:FindFirstChild("Area")
                        
                        if areaFolder then
                            for _, mapFolder in ipairs(areaFolder:GetChildren()) do
                                local serverZone = mapFolder:FindFirstChild("ServerZone")
                                local recoverFolder = serverZone and serverZone:FindFirstChild("Recover")
                                
                                if recoverFolder then
                                    for _, recPart in ipairs(recoverFolder:GetDescendants()) do
                                        if recPart:IsA("BasePart") then
                                            local dist = (hrp.Position - recPart.Position).Magnitude
                                            if dist < minDist then
                                                minDist = dist
                                                bestRecoverPos = recPart.Position
                                            end
                                        end
                                    end
                                end
                            end
                        end
                        
                        if bestRecoverPos then
                            print("🚀 [Auto Heal] Đang bay đến trạm hồi máu gần nhất...")
                            hrp.CFrame = CFrame.new(bestRecoverPos + Vector3.new(0, 3, 0))
                            task.wait(1)
                            
                            print("⏳ [Auto Heal] Đang chờ Pet hồi sinh (Tối đa 30s)...")
                            task.wait(8) -- Chờ 8s mặc định để game xử lý
                            
                            -- Vòng lặp chờ chắc chắn Pet đã sống lại
                            local startTime = os.time()
                            local isStillDead = true
                            
                            while (os.time() - startTime < 30) and camSettings.autoHeal do
                                isStillDead = false
                                for _, pet in pairs(petsFolder:GetChildren()) do
                                    if string.find(pet.Name, myId) then
                                        local hpP = pet:FindFirstChild("Health")
                                        if hpP then
                                            local cHp = tonumber(string.split(tostring(hpP.Value), ",")[1])
                                            if cHp and cHp <= 0 then
                                                isStillDead = true
                                                break
                                            end
                                        end
                                    end
                                end
                                
                                if not isStillDead then break end
                                task.wait(0.5)
                            end
                            
                            print("✅ [Auto Heal] Đã hồi máu xong! Bay về bãi farm...")
                            if camSettings.savedPosition then
                                hrp.CFrame = camSettings.savedPosition
                            end
                        else
                            print("⚠️ [Auto Heal] Không tìm thấy bục Recover trên map hiện tại!")
                        end
                    end
                    
                    -- Reset trạng thái để tiếp tục farm
                    camSettings.isHealing = false
                    camSettings.savedPosition = nil
                    print("🔄 [Auto Heal] Hoàn tất chu trình.")
                end
            end)
            
            -- Lớp phòng thủ cuối cùng: Nếu lỡ lỗi văng ra ngoài pcall, tự động gỡ kẹt
            if camSettings.isHealing and not hasDead then
                camSettings.isHealing = false
            end
        end
    end
end)

local function SetupChat(p) 
    p.Chatted:Connect(function(msg) 
        if msg == "!f" then 
            targetPlayer = p 
            following = true 
            StatusLabel:Set("Trạng thái: Đang theo dõi " .. p.Name) 
        end 
    end) 
end

for _, p in ipairs(Players:GetPlayers()) do 
    if p ~= LocalPlayer then 
        SetupChat(p) 
    end 
end
Players.PlayerAdded:Connect(SetupChat)

local mainCoroutine = coroutine.create(function()
    while task.wait(0.5) do
        -- 1. Logic Follow
        if following and targetPlayer then
            pcall(function()
                local char = LocalPlayer.Character
                local tChar = targetPlayer.Character
                
                if char and tChar and tChar:FindFirstChild("HumanoidRootPart") then
                    missingTime = 0
                    char.Humanoid:MoveTo(tChar.HumanoidRootPart.Position)
                else
                    if missingTime == 0 then 
                        missingTime = os.time() 
                    end
                    
                    if os.time() - missingTime >= 30 and tChar and tChar:FindFirstChild("HumanoidRootPart") then
                        char.HumanoidRootPart.CFrame = tChar.HumanoidRootPart.CFrame
                        missingTime = 0
                    end
                end
            end)
        end

        -- 2. Logic Auto Attack Dungeon (Bằng click UI)
        if autoAttackEnabled then
            pcall(function()
                local mainGui = playerGui:FindFirstChild("MainGui")
                if mainGui then
                    local screenGui = mainGui:FindFirstChild("ScreenGui")
                    if screenGui and screenGui:FindFirstChild("MainBottomView") then
                        local fmAutoAttack = screenGui.MainBottomView:FindFirstChild("FmAutoAttack")
                        if fmAutoAttack and fmAutoAttack:FindFirstChild("BtAutoOn") and fmAutoAttack.BtAutoOn.Visible and fmAutoAttack.BtAutoOn.AbsoluteSize.X > 0 then
                            ClickButtonExact(fmAutoAttack.BtAutoOn, "Nút Bật Đánh")
                        end
                    end
                end
            end)
        end

        -- 3. Logic Auto Join & LeaveAth
        local labStage = findUIElementByName(playerGui, "LabStage")
        
        if autoJoinOthersEnabled and (not labStage or labStage.Text == "Unknown") then
            pcall(function()
                local wArea = workspace:FindFirstChild("Area")
                local center = wArea and wArea:FindFirstChild("center")
                local innerArea = center and center:FindFirstChild("Area")
                local abyssFolder = innerArea and innerArea:FindFirstChild("Abyss")
                
                if abyssFolder then
                    local char = LocalPlayer.Character
                    local hrp = char and char:FindFirstChild("HumanoidRootPart")
                    local humanoid = char and char:FindFirstChild("Humanoid")
                    
                    if hrp and humanoid and humanoid.Health > 0 then
                        local alreadyInSomeoneRoom = false
                        local shouldJumpOut = false
                        local jumpOutTarget = nil
                        local roomToJoin = nil
                        
                        for _, room in ipairs(abyssFolder:GetChildren()) do
                            if string.match(room.Name, "Abyss_") then
                                local platform = room:FindFirstChild("Platform")
                                local board = platform and platform:FindFirstChild("Board")
                                local surfaceGui = board and board:FindFirstChild("SurfaceGui")
                                
                                if surfaceGui and surfaceGui:FindFirstChild("FmEmpty") and surfaceGui:FindFirstChild("FmPrepare") and surfaceGui:FindFirstChild("FmInfo") and surfaceGui:FindFirstChild("FmPlayer") then
                                    if (not surfaceGui.FmEmpty.Visible) and (not surfaceGui.FmPrepare.Visible) and surfaceGui.FmInfo.Visible and surfaceGui.FmPlayer.Visible then
                                        
                                        local isOtherPlayer = false
                                        local isMyRoom = false
                                        
                                        for _, desc in ipairs(surfaceGui.FmPlayer:GetDescendants()) do
                                            if desc:IsA("ImageLabel") and string.find(desc.Image, "AvatarHeadShot") then
                                                local extractedId = string.match(desc.Image, "id=(%d+)")
                                                if extractedId then 
                                                    if tonumber(extractedId) ~= myUserId then 
                                                        isOtherPlayer = true 
                                                    else 
                                                        isMyRoom = true 
                                                    end 
                                                    break 
                                                end
                                            end
                                        end
                                        
                                        local targetPos = (room:IsA("Model") and room:GetPivot().Position) or (room:IsA("BasePart") and room.Position)
                                        
                                        if targetPos then
                                            local dist = (hrp.Position - targetPos).Magnitude
                                            
                                            if isMyRoom and dist < 25 then
                                                shouldJumpOut = true
                                                local leaveAth = room:FindFirstChild("LeaveAth")
                                                if leaveAth and leaveAth:IsA("Attachment") then 
                                                    jumpOutTarget = leaveAth.WorldPosition 
                                                end
                                            elseif isOtherPlayer then
                                                if dist < 15 then 
                                                    alreadyInSomeoneRoom = true 
                                                elseif not roomToJoin then 
                                                    roomToJoin = targetPos + Vector3.new(0, 5, 0) 
                                                end
                                            end
                                        end
                                    end
                                end
                            end
                        end
                        
                        if shouldJumpOut then
                            if jumpOutTarget then 
                                hrp.CFrame = CFrame.new(jumpOutTarget + Vector3.new(0, 3, 0)) 
                            else 
                                hrp.CFrame = hrp.CFrame * CFrame.new(0, 0, 20) 
                            end
                            task.wait(2)
                        elseif roomToJoin and not alreadyInSomeoneRoom then
                            hrp.CFrame = CFrame.new(roomToJoin)
                            task.wait(3)
                        end
                    end
                end
            end)
        end

        -- 4. Logic Auto-Exit Hầm Ngục
        if autoExitEnabled and not autoExitTriggered then
            local fmTimeFrame = nil
            pcall(function() 
                fmTimeFrame = playerGui.MainGui.ScreenGui.ArenaMainRightTopView.FmTime 
            end)
            
            if labStage then
                local currentStage, maxStage = string.match(labStage.Text, "(%d+)/(%d+)")
                
                if currentStage and maxStage and tonumber(currentStage) == tonumber(maxStage) then
                    
                    if fmTimeFrame and fmTimeFrame.Visible == true then
                        if not stage20ClockStarted then 
                            stage20ClockStarted = true 
                            DungeonStatus:Set("Trạng thái Auto Exit: Đồng hồ màn 20 đã chạy...") 
                        end
                    end
                    
                    if stage20ClockStarted and (not fmTimeFrame or fmTimeFrame.Visible == false) then
                        autoExitTriggered = true
                        DungeonStatus:Set("Trạng thái Auto Exit: Boss đã gục, chuẩn bị click thoát...")
                        
                        task.spawn(function() 
                            SendDiscordNotification("**" .. LocalPlayer.Name .. "** vừa hoàn thành một vòng Dungeon 20/20!") 
                        end)
                        
                        task.spawn(SafeExitRoutine)
                    end
                else
                    stage20ClockStarted = false
                    autoExitTriggered = false
                end
            end
        end
    end
end)
coroutine.resume(mainCoroutine)

coordText.BackgroundTransparency = 1
coordText.TextColor3 = Color3.fromRGB(0, 255, 0) -- Đổi sang màu xanh lá cho dễ nhìn
coordText.Font = Enum.Font.Code
coordText.TextSize = 16
coordText.TextXAlignment = Enum.TextXAlignment.Left
coordText.TextYAlignment = Enum.TextYAlignment.Top
coordText.Text = "0 / 0 / 0"
coordText.Active = true

-- Cập nhật tọa độ liên tục
local RunService = game:GetService("RunService")
RunService.RenderStepped:Connect(function()
    local char = game.Players.LocalPlayer.Character
    if char and char:FindFirstChild("HumanoidRootPart") then
        local p = char.HumanoidRootPart.Position
        coordText.Text = string.format("📍 %.1f / %.1f / %.1f", p.X, p.Y, p.Z)
    end
end)

-- Xử lý Kéo thả (Drag)
local UserInputService = game:GetService("UserInputService")
local dragging = false
local dragStart = nil
local startPos = nil

coordText.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = coordText.Position
    end
end)

coordText.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if dragging and (input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseMovement) then
        local delta = input.Position - dragStart
        coordText.Position = UDim2.new(
            startPos.X.Scale,
            startPos.X.Offset + delta.X,
            startPos.Y.Scale,
            startPos.Y.Offset + delta.Y
        )
    end
end)
