local function lEncode(data)  return HttpService:JSONEncode(data)  end
local function lDecode(data)  return HttpService:JSONDecode(data)  end

local function lDigest(input)
    local inputStr = tostring(input)
    local hash = {}
    for i = 1, #inputStr do
        table.insert(hash, string.byte(inputStr, i))
    end
    local hashHex = ""
    for _, byte in ipairs(hash) do
        hashHex = hashHex .. string.format("%02x", byte)
    end
    return hashHex
end

local function cacheLink()
    if not fRequest then
        return false, "Executor not supported (HTTP request API missing)."
    end
    if cachedTime + (10*60) < fOsTime() then
        local success, response = pcall(function()
            return fRequest({
                Url    = host .. "/public/start",
                Method = "POST",
                Body   = lEncode({
                    service    = service,
                    identifier = lDigest(fGetHwid())
                }),
                Headers = { ["Content-Type"] = "application/json" }
            })
        end)
        if not success then
            return false, "pcall error: " .. tostring(response)
        end
        if not response then
            return false, "Response from request is nil."
        end
        if response.StatusCode == 200 then
            local decoded = lDecode(response.Body)
            if decoded.success == true then
                cachedLink = decoded.data.url
                cachedTime = fOsTime()
                return true, cachedLink
            else
                return false, decoded.message
            end
        elseif response.StatusCode == 429 then
            return false, "Rate limited. Wait 20 seconds."
        end
        return false, "Failed to get link."
    else
        return true, cachedLink
    end
end

end

local function redeemKey(key)
    if not fRequest then
        return false, "Executor not supported (HTTP request API missing)."
    end
    local nonce    = generateNonce()
    local endpoint = host .. "/public/redeem/" .. fToString(service)
    local body     = { identifier = lDigest(fGetHwid()), key = key }
    if useNonce then body.nonce = nonce end

    local success, response = pcall(function()
        return fRequest({
            Url     = endpoint,
            Method  = "POST",
            Body    = lEncode(body),
            Headers = { ["Content-Type"] = "application/json" }
        })
    end)

    if not success then
        return false, "pcall error: " .. tostring(response)
    end
    if not response then
        return false, "Response from request is nil."
    end

    if response.StatusCode == 200 then
        local decoded = lDecode(response.Body)
        if decoded.success == true then
            if decoded.data.valid == true then
                if useNonce then
                    if decoded.data.hash == lDigest("true" .. "-" .. nonce .. "-" .. secret) then
                        return true
                    else
                        return false, "failed to verify integrity."
                    end
                else
                    return true
                end
            else
                return false, "key is invalid."
            end
        else
            if fStringSub(decoded.message, 1, 27) == "unique constraint violation" then
                return false, "you already have an active key."
            else
                return false, decoded.message
            end
        end
    elseif response.StatusCode == 429 then
        return false, "Rate limited. Wait 20 seconds."
    else
        return false, "Server error. Try again later."
    end
end

local function verifyKey(key)
    if not fRequest then
        return false, "Executor not supported (HTTP request API missing)."
    end
    if requestSending then
        return false, "A request is already sending."
    end
    requestSending = true

    local nonce    = generateNonce()
    local endpoint = host .. "/public/whitelist/" .. fToString(service)
        .. "?identifier=" .. lDigest(fGetHwid()) .. "&key=" .. key
    if useNonce then
        endpoint = endpoint .. "&nonce=" .. nonce
    end

    local success, response = pcall(function()
        return fRequest({ Url = endpoint, Method = "GET" })
    end)
    requestSending = false

    if not success then
        return false, "pcall error: " .. tostring(response)
    end
    if not response then
        return false, "Response from request is nil."
    end

    if response.StatusCode == 200 then
        local decoded = lDecode(response.Body)
        if decoded.success == true then
            if decoded.data.valid == true then
                return true
            else
                    return redeemKey(key)
            end
        else
            return false, decoded.message
        end
    elseif response.StatusCode == 429 then
        return false, "Rate limited. Wait 20 seconds."
    else
        return false, "Server error. Try again later."
    end
end

-- ── VERIFY ──────────────────────────────────────
local busy = false
vBtn.MouseButton1Click:Connect(function()
    if busy then return end; busy = true

    local entered = keyTB.Text:match("^%s*(.-)%s*$")
    statusLbl.TextTransparency = 0
    statusLbl.Text = ""; statusLbl.TextColor3 = Color3.fromRGB(255,55,55)

    if entered == "" then
        statusLbl.Text = "Please enter your key"
        busy = false; return
    end

    -- Animated dots while checking
    local dots, conn = 0, nil
    conn = RunService.Heartbeat:Connect(function()
        dots=(dots+1)%4; vBtnLbl.Text="CHECKING"..string.rep(".",dots)
    end)

    -- Run PlatoBoost verification in a separate thread so UI stays responsive
    local valid, errMsg
    task.spawn(function()
        valid, errMsg = verifyKey(entered)
    end)
    -- Wait for verifyKey to finish (max 10 seconds)
    local waited = 0
    while valid == nil and waited < 10 do
        task.wait(0.1)
        waited = waited + 0.1
    end

    conn:Disconnect()
    vBtnLbl.Text = "VERIFY KEY"

    if valid == true then
        if writefile then
            pcall(function() writefile(key_file, entered) end)
        end
        doSuccess()
    elseif waited >= 10 then
        statusLbl.TextColor3 = Color3.fromRGB(255,180,0)
        statusLbl.Text = "Server timeout. Try again"
        warn("[KeySystem] Verification timed out (10s)")
        doShake(); busy = false
    else
        doFail(errMsg or "Invalid or expired key")
        warn("[KeySystem] Verification failed: " .. tostring(errMsg))
        busy = false
    end
end)

-- ── AUTO LOGIN CHECK ────────────────────────────
task.spawn(function()
    if isfile and readfile and isfile(key_file) then
        local cachedKey = readfile(key_file)
        if cachedKey and cachedKey:match("%S") then
            cachedKey = cachedKey:match("^%s*(.-)%s*$")
            keyTB.Text = cachedKey
            
            -- Wait a brief moment for intro animation to complete/progress
            task.wait(1.5)
            if not busy then
                busy = true
                statusLbl.TextTransparency = 0
                statusLbl.Text = "Checking cached key..."
                statusLbl.TextColor3 = Color3.fromRGB(100,150,255)
                
                local dots, conn = 0, nil
                conn = RunService.Heartbeat:Connect(function()
                    dots=(dots+1)%4; vBtnLbl.Text="CHECKING"..string.rep(".",dots)
                end)
                
                local valid, errMsg
                task.spawn(function()
                    valid, errMsg = verifyKey(cachedKey)
                end)
                
                local waited = 0
                while valid == nil and waited < 10 do
                    task.wait(0.1)
                    waited = waited + 0.1
                end
                
                conn:Disconnect()
                vBtnLbl.Text = "VERIFY KEY"
                
                if valid == true then
                    doSuccess()
                else
                    busy = false
                    statusLbl.Text = "Cached key expired or invalid"
                    statusLbl.TextColor3 = Color3.fromRGB(255,180,0)
                    keyTB.Text = ""
                    if delfile then pcall(function() delfile(key_file) end) end
                end
            end
        end
    end
end)
