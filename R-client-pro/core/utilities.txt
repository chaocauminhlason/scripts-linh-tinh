local Utils = {}

local Workspace = game:GetService("Workspace")

-- ==========================================
-- 1. HÀM QUÉT UI ĐỆ QUY (Tuyệt đối)
-- ==========================================
function Utils.FindUIElementByName(parent, name)
    if not parent then return nil end
    for _, child in ipairs(parent:GetDescendants()) do
        if child.Name == name then 
            return child 
        end
    end
    return nil
end

-- ==========================================
-- 2. HÀM CLICK CHUỘT ẢO (Bù trừ GuiInset chống trượt)
-- ==========================================
function Utils.ClickButtonExact(button, debugName)
    if not button then return end
    
    local absPos = button.AbsolutePosition
    local absSize = button.AbsoluteSize
    
    local centerX = absPos.X + (absSize.X / 2)
    local finalY = absPos.Y + (absSize.Y / 2)
    
    local screenGui = button:FindFirstAncestorOfClass("ScreenGui")
    if screenGui and not screenGui.IgnoreGuiInset then 
        finalY = finalY + GuiService:GetGuiInset().Y 
    end
    
    VirtualInputManager:SendMouseButtonEvent(centerX, finalY, 0, true, game, 1)
    task.wait(0.1)
    VirtualInputManager:SendMouseButtonEvent(centerX, finalY, 0, false, game, 1)
    
    if debugName then
        print("🖱️ [Utils] Đã click ảo vào: " .. debugName)
    end
end

-- ==========================================
-- 3. HÀM LẤY ID NGƯỜI CHƠI (Hệ thống R-Game)
-- ==========================================
local cachedPlayerId = nil
function Utils.GetPlayerId()
    if cachedPlayerId then 
        return cachedPlayerId 
    end
    
    local petsFolder = Workspace:FindFirstChild("Pets")
    if not petsFolder then return nil end
    
    for _, pet in pairs(petsFolder:GetChildren()) do
        if pet:IsA("BasePart") then
            local playerId = pet:GetAttribute("PlayerId")
            if playerId then
                cachedPlayerId = playerId
                return playerId
            end
        end
    end
    
    return nil
end

-- HÀM X-RAY QUÉT MẶT ĐẤT
local function GetGroundPosition(targetPosition)
    -- 1. Đặt mốc phát tia: Từ vị trí mục tiêu, ta nâng lên cao 50 studs (trên trời)
    local rayOrigin = targetPosition + Vector3.new(0, 50, 0)
    
    -- 2. Hướng bắn tia: Bắn thẳng xuống dưới mặt đất sâu 100 studs
    local rayDirection = Vector3.new(0, -100, 0)

    -- 3. Cấu hình X-ray: Bộ lọc để tia này xuyên qua nhân vật và thú cưỡi (chỉ chạm đất)
    local raycastParams = RaycastParams.new()
    raycastParams.FilterType = Enum.RaycastFilterType.Exclude
    
    -- Lọc bỏ Character của mình và thư mục Pets (thêm thư mục Pets tùy game)
    local char = game.Players.LocalPlayer.Character
    local petsFolder = game.Workspace:FindFirstChild("Pets")
    raycastParams.FilterDescendantsInstances = {char, petsFolder} 

    -- 4. Thực hiện quét
    local raycastResult = game.Workspace:Raycast(rayOrigin, rayDirection, raycastParams)

    if raycastResult then
        -- TRÚNG ĐÍCH: Trả về tọa độ mặt đất + bù thêm 3 studs chiều cao của chân nhân vật
        return raycastResult.Position + Vector3.new(0, 3, 0)
    else
        -- NẾU TRƯỢT (Rất hiếm): Trả về tọa độ gốc để phòng hờ
        return targetPosition + Vector3.new(0, 3, 0)
    end
end

-- ==============================================================
    -- HỆ THỐNG CHIẾN ĐẤU & TƯƠNG TÁC REMOTE (DÙNG CHUNG)
    -- ==============================================================
    
    -- 1. Lên / Xuống thú cưỡi
    Utils.ToggleMount = function(DataPullFunc, state)
        pcall(function() DataPullFunc:InvokeServer("PetSwitchRideStatusChannel", state) end)
    end

    -- 2. Tự động nhảy xuống thú an toàn
    Utils.SmartDismount = function(LocalPlayer, DataPullFunc)
        local isRiding = LocalPlayer:GetAttribute("RidePetId") ~= nil
        if isRiding then
            Utils.ToggleMount(DataPullFunc, false)
            task.wait(0.3)
            local char = LocalPlayer.Character
            local humanoid = char and char:FindFirstChild("Humanoid")
            if humanoid then humanoid.Jump = true end
            task.wait(0.1)
        end
    end

    -- 3. Gửi lệnh tấn công quái
    Utils.AttackMonster = function(DataPullFunc, monsterId)
        pcall(function() DataPullFunc:InvokeServer("MonsterAttackChannel", monsterId) end)
    end

    -- 4. Gửi lệnh ném bóng bắt quái
    Utils.CatchMonster = function(DataPullFunc, monsterId)
        pcall(function() DataPullFunc:InvokeServer("MonsterCatchStartChannel", monsterId) end)
    end

    -- 5. Gửi lệnh Rời Phòng / Thoát Trận
    Utils.LeaveMatch = function(DataPullFunc, riftId, teamId)
        pcall(function()
            DataPullFunc:InvokeServer("ArenaLeaveChannel")
            if riftId and teamId then
                DataPullFunc:InvokeServer("DungeonLeaveTeamChannel", riftId, teamId)
            end
        end)
    end

    -- 6. HỆ THỐNG RADAR: Quét, phân loại và tìm quái vật gần nhất
    Utils.ScanMonsters = function(monstersFolder, hrp)
        local hasAlive = false
        local deadList = {}
        local bestTarget = nil
        local closestDist2D = math.huge

        if monstersFolder then
            for _, m in pairs(monstersFolder:GetChildren()) do
                if m:IsA("Model") or m:IsA("BasePart") then
                    local hum = m:FindFirstChildOfClass("Humanoid")
                    if hum then
                        if hum.Health <= 0 then
                            table.insert(deadList, m) -- Quái trắng máu để Catch
                        else
                            hasAlive = true
                            local pos = m:IsA("Model") and m.PrimaryPart and m.PrimaryPart.Position or m.Position
                            local dist2D = math.sqrt((hrp.Position.X - pos.X)^2 + (hrp.Position.Z - pos.Z)^2)
                            if dist2D < closestDist2D then closestDist2D = dist2D; bestTarget = m end
                        end
                    else
                        hasAlive = true
                        local pos = m:IsA("Model") and m.PrimaryPart and m.PrimaryPart.Position or m.Position
                        local dist2D = math.sqrt((hrp.Position.X - pos.X)^2 + (hrp.Position.Z - pos.Z)^2)
                        if dist2D < closestDist2D then closestDist2D = dist2D; bestTarget = m end
                    end
                end
            end
        end

        return hasAlive, deadList, bestTarget, closestDist2D
    end

-- Trả về bảng Utils chứa tất cả các hàm trên để các file khác gọi lại
return Utils
