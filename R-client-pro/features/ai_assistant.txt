        if isfile and isfile(API_KEY_FILE) then
            local key = readfile(API_KEY_FILE)
            if key and key ~= "" then
                return string.gsub(key, "^%s*(.-)%s*$", "%1")
            end
        end
        return nil
    end

        if writefile then
            pcall(function() writefile(API_KEY_FILE, key) end)

    end

    local function RemoveAccents(str)
        if type(str) ~= "string" then return str end
        for k, v in pairs(UTF8_MAP) do
            str = string.gsub(str, k, v)
        end
        return str
    end

        rawCode = string.gsub(rawCode, "```lua\n?", "")
        rawCode = string.gsub(rawCode, "```\n?", "")
        rawCode = string.gsub(rawCode, "^%s*(.-)%s*$", "%1")
        return rawCode

    -- ==========================================

    local function TrySendWebhookLog(prompt, logMsg)
        if not (useWebhook and WebhookModule and type(WebhookModule.SendNotification) == "function") then return end
        local finalLog = RemoveAccents("[AI Assistant Log]\nUser Prompt: " .. prompt .. "\n" .. logMsg)
        pcall(function() WebhookModule.SendNotification(finalLog) end)
    end

    -- ==========================================
    -- HÀM GỌI GEMINI API
    -- ==========================================

    local function RequestAICode(userPrompt)
        local savedKey = GetSavedAPIKey()
        if not savedKey then
            return false, "Chưa có API Key."
        end

        local reqFunc = request or http_request or (syn and syn.request)
        if not reqFunc then
            return false, "Executor không hỗ trợ HTTP Request!"
        end

        local API_URL = "https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=" .. savedKey
        local systemInstruction = "You are a Roblox Lua exploit developer. Only respond with raw, executable Lua code. Do not include markdown blocks, do not include ```lua tags, no explanations."
        local payload = {
            contents = { { parts = { { text = systemInstruction .. "\nUser Request: " .. userPrompt } } } }
        }

        local success, response = pcall(function()
            return reqFunc({
                Url = API_URL,
                Method = "POST",
                Headers = { ["Content-Type"] = "application/json" },
                Body = HttpService:JSONEncode(payload)
            })
        end)

        if not success then
            return false, "Lỗi kết nối API."
        end

        if response.StatusCode ~= 200 then
            return false, "HTTP " .. tostring(response.StatusCode)
        end

        local ok, data = pcall(function() return HttpService:JSONDecode(response.Body) end)
        if not ok then
            return false, "Lỗi parse JSON từ API."
        end

        local part = data.candidates
            and data.candidates[1]
            and data.candidates[1].content
            and data.candidates[1].content.parts
            and data.candidates[1].content.parts[1]

        if not part then
            return false, "API trả về response rỗng."
        end

        return true, part.text
    end

    -- HÀM THỰC THI CODE (tách riêng để dùng lại)
    -- ==========================================

    local function ExecuteCode(code, prompt)
        local logMsg = ""

        local func, compileError = loadstring(code)
        if not func then
            logMsg = "Compile Error:\n" .. tostring(compileError)
            OutputLabel.Text = OutputLabel.Text .. "\n\n❌ LỖI CÚ PHÁP:\n" .. tostring(compileError)
            TrySendWebhookLog(prompt or "", logMsg)
            return
        end

        local runSuccess, runtimeError = pcall(func)
        if runSuccess then
            logMsg = "Code executed successfully:\n" .. code
            OutputLabel.Text = OutputLabel.Text .. "\n\n✅ THỰC THI THÀNH CÔNG"
        else
            logMsg = "Runtime Error:\n" .. tostring(runtimeError)
            OutputLabel.Text = OutputLabel.Text .. "\n\n❌ LỖI RUNTIME:\n" .. tostring(runtimeError)
        end

        TrySendWebhookLog(prompt or "", logMsg)
        ResetExecuteButton()
    end

                SubmitBtn.Text = "❌ LỖI"

            end

            isProcessing = false

            task.wait(2)

            if success then
                SubmitBtn.Text = "🤖 LẤY CODE TỪ AI"
                SubmitBtn.BackgroundColor3 = Color3.fromRGB(0, 120, 215)
            else
                ResetSubmitButton()
            end
        end)
