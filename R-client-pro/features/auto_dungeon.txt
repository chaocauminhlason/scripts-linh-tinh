-- ====================================================================
-- MODULE: DUNGEON AUTO (TÍCH HỢP TỐI ƯU PHÍM M, AUTO EXIT & AUTO DUNGEON)
-- BẢN CẬP NHẬT: THÊM CÔNG TẮC TỔNG ĐIỀU KHIỂN TOÀN BỘ CHỨC NĂNG CON
-- ====================================================================
return function(Window, Utils, WebhookModule)
    local Players = game:GetService("Players")
    local Workspace = game:GetService("Workspace")
    local ReplicatedStorage = game:GetService("ReplicatedStorage")
    local VirtualInputManager = game:GetService("VirtualInputManager")
    
    local LocalPlayer = Players.LocalPlayer
    local playerGui = LocalPlayer:WaitForChild("PlayerGui")
    local myUserId = LocalPlayer.UserId

    -- Biến trạng thái hệ thống
    local masterSwitchEnabled = false -- [MỚI] AUTO DUNGEON CỦA MODULE
    local autoExitEnabled = false
    local autoExitTriggered = false
    local stage20ClockStarted = false
    local autoAttackEnabled = false
    local autoJoinOthersEnabled = false
    
    local dungeonStuckTimer = 0
    local dungeonLastPosition = nil

    -- API Game
    local MonsterSystem = nil
    pcall(function() MonsterSystem = require(ReplicatedStorage.CommonLogic.Monster.MonsterSystem) end)

    -- ==========================================
    -- GIAO DIỆN UI (RAYFIELD TABS)
    -- ==========================================
    local DungeonTab = Window:CreateTab("Hầm Ngục", 4483362458)
    local DungeonStatus = DungeonTab:CreateLabel("🔌 Trạng thái hệ thống: AUTO DUNGEON ĐANG TẮT")

    -- [MỚI] AUTO DUNGEON - PHẢI BẬT NÚT NÀY THÌ CÁC CHỨC NĂNG CON MỚI CÓ TÁC DỤNG
    DungeonTab:CreateToggle({
        Name = "Auto dungeon",
        CurrentValue = false,
        Flag = "DungeonMasterBreaker",
        Callback = function(Value)
            masterSwitchEnabled = Value
            if not Value then
                DungeonStatus:Set("🔌 Trạng thái hệ thống: AUTO DUNGEON ĐANG TẮT")
                -- Ép dừng nhân vật ngay lập tức khi sập cầu dao để tránh bị chạy tự do
                pcall(function()
                    local char = LocalPlayer.Character
                    if char and char:FindFirstChild("Humanoid") then
                        char.Humanoid:MoveTo(char.HumanoidRootPart.Position)
                    end
                end)
            else
                DungeonStatus:Set("⚡ Hệ thống đã cấp điện! Các nút con bắt đầu hoạt động.")
            end
        end
    })

    DungeonTab:CreateSection("--- CÁC CHỨC NĂNG CON VẬN HÀNH ---")

    DungeonTab:CreateToggle({ 
        Name = "Tự Động Tìm & Đánh Quái (Tối Ưu Phím M)", 
        CurrentValue = false, 
        Flag = "AutoAttack_V2", 
        Callback = function(Value) 
            autoAttackEnabled = Value 
            if not Value then
                pcall(function()
                    local char = LocalPlayer.Character
                    if char and char:FindFirstChild("Humanoid") then
                        char.Humanoid:MoveTo(char.HumanoidRootPart.Position)
                    end
                end)
            end
        end 
    })

    DungeonTab:CreateToggle({ 
        Name = "Tự Động Thoát Khi Xong Màn 20", 
        CurrentValue = false, 
        Flag = "AutoExitDungeon", 
        Callback = function(Value) 
            autoExitEnabled = Value 
            if Value then 
                if masterSwitchEnabled then DungeonStatus:Set("Trạng thái Auto Exit: ĐANG BẬT VÀ QUÉT...") end
            else 
                autoExitTriggered = false
                stage20ClockStarted = false 
            end 
        end 
    })

    DungeonTab:CreateToggle({ 
        Name = "Tự Động Nhảy Vào Phòng Người Khác", 
        CurrentValue = false, 
        Flag = "AutoJoinOthers", 
        Callback = function(Value) autoJoinOthersEnabled = Value end 
    })

    -- LOGIC QUÉT UI THOÁT
    local function SafeExitRoutine()
        local exitBtn = Utils.FindUIElementByName(playerGui, "BgLeave")
        if exitBtn and exitBtn.AbsoluteSize.X > 0 then Utils.ClickButtonExact(exitBtn, "Nút Thoát") else autoExitTriggered = false return end
        
        task.wait(1.5)
        local mainGui = playerGui:FindFirstChild("MainGui")
        if mainGui and mainGui:FindFirstChild("ScreenGui") and mainGui.ScreenGui:FindFirstChild("ConfirmView") then
            local fmBottom = mainGui.ScreenGui.ConfirmView:FindFirstChild("FmBottom")
            if fmBottom and fmBottom:FindFirstChild("BtOk") and fmBottom.BtOk.AbsoluteSize.X > 0 then
                for i = 1, 3 do Utils.ClickButtonExact(fmBottom.BtOk, "Nút Yes") task.wait(0.2) end
            else autoExitTriggered = false return end
        else autoExitTriggered = false return end
        
        task.wait(2.5)
        local checkLabStage = Utils.FindUIElementByName(playerGui, "LabStage")
        local currentStageText = checkLabStage and checkLabStage.Text or "Unknown"
        if currentStageText == "Unknown" or not string.match(currentStageText, "20/20") then 
            autoExitTriggered = false; stage20ClockStarted = false 
        else autoExitTriggered = false end
    end

    -- ==========================================
    -- VÒNG LẶP CORE (CHỈ CHẠY KHI AUTO DUNGEON BẬT)
    -- ==========================================
    task.spawn(function()
        while task.wait(0.5) do
            -- [MỚI] KIỂM TRA AUTO DUNGEON ĐẦU TIÊN
            if not masterSwitchEnabled then continue end

            local char = LocalPlayer.Character
            local hrp = char and char:FindFirstChild("HumanoidRootPart")
            local humanoid = char and char:FindFirstChild("Humanoid")
           
            if not (hrp and humanoid and humanoid.Health > 0) then continue end
            local labStage = Utils.FindUIElementByName(playerGui, "LabStage")
            
            -- [1] LOGIC AUTO ATTACK (TRICK PHÍM M)
            if autoAttackEnabled and labStage and labStage.Text ~= "Unknown" and labStage.Text ~= "" then
                local monstersFolder = Workspace:FindFirstChild("Monsters")
                local closestDist2D = math.huge
                local bestTarget = nil

                if monstersFolder then
                    for _, monster in pairs(monstersFolder:GetChildren()) do
                        if monster:IsA("Model") or monster:IsA("BasePart") then
                            local hum = monster:FindFirstChildOfClass("Humanoid")
                            if (not hum) or (hum and hum.Health > 0) then
                                local pos = monster:IsA("Model") and monster.PrimaryPart and monster.PrimaryPart.Position or monster.Position
                                local dist2D = math.sqrt((hrp.Position.X - pos.X)^2 + (hrp.Position.Z - pos.Z)^2)
                                
                                if dist2D < closestDist2D then 
                                    closestDist2D = dist2D
                                    bestTarget = monster 
                                end
                            end
                        end
                    end
                end

                if bestTarget then
                    local targetPos = bestTarget:IsA("Model") and bestTarget.PrimaryPart and bestTarget.PrimaryPart.Position or bestTarget.Position
                    
                    if closestDist2D > 40 then
                        -- TỐC BIẾN KHI Ở XA
                        VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.M, false, game) task.wait() VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.M, false, game) task.wait(0.4)
                        hrp.CFrame = CFrame.new(targetPos + Vector3.new(0, 0, 4)) task.wait(0.2)
                        humanoid:MoveTo(hrp.Position) dungeonLastPosition = hrp.Position dungeonStuckTimer = 0
                        
                    elseif closestDist2D > 12 then
                        -- CHẠY/BAY LẠI GẦN
                        humanoid:MoveTo(targetPos)
                        if dungeonLastPosition and (hrp.Position - dungeonLastPosition).Magnitude < 1 then
                            dungeonStuckTimer = dungeonStuckTimer + 0.5
                            if dungeonStuckTimer >= 3 then hrp.CFrame = CFrame.new(targetPos + Vector3.new(0, 0, 3)) dungeonStuckTimer = 0 end
                        else dungeonStuckTimer = 0 end
                        dungeonLastPosition = hrp.Position
                        
                    else
                        -- ĐỦ TẦM ĐÁNH (< 12)
                        humanoid:MoveTo(hrp.Position)
                        dungeonStuckTimer = 0
                        

                    -- FIX KẸT THÚ: Ép nhân vật rời khỏi thú cưỡi ngay khi áp sát quái
                    if humanoid.Sit then
                        humanoid.Sit = false
                        task.wait(0.1)
    
                        -- Giả lập bấm phím M để báo cho Server game biết đã cất thú
                        VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.M, false, game) 
                        task.wait() 
                        VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.M, false, game)
    
                        task.wait(0.15)
                        humanoid.Jump = true
                        task.wait(0.15) -- Giữ một nhịp cực nhỏ để server nhận diện trạng thái đáp đất an toàn
                    end
                        
                        local id = tonumber(bestTarget.Name:match("Monster_(%d+)"))
                        if id and MonsterSystem then 
                            MonsterSystem.ClientAttackMonster(id) 
                        end
                    end
                else 
                    -- KHÔNG TÌM THẤY QUÁI (PHÒNG TRỐNG)
                    dungeonStuckTimer = 0 
                    
                    if humanoid.Sit then
                        humanoid.Sit = false
                        task.wait(0.1)
                        VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.M, false, game) 
                        task.wait() 
                        VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.M, false, game)
                        task.wait(0.2)
                        humanoid.Jump = true 
                    end
                end
            end

            -- [2] LOGIC AUTO EXIT 
            if autoExitEnabled and not autoExitTriggered and labStage then
                local fmTimeFrame = nil
                pcall(function() fmTimeFrame = playerGui.MainGui.ScreenGui.ArenaMainRightTopView.FmTime end)
                local currentStage, maxStage = string.match(labStage.Text, "(%d+)/(%d+)")
                
                if currentStage and maxStage and tonumber(currentStage) == tonumber(maxStage) then
                    if fmTimeFrame and fmTimeFrame.Visible == true then
                        if not stage20ClockStarted then 
                            stage20ClockStarted = true
                            DungeonStatus:Set("Trạng thái Auto Exit: Đồng hồ màn 20 đã chạy...") 
                        end
                    end
                    if stage20ClockStarted and (not fmTimeFrame or fmTimeFrame.Visible == false) then
                        autoExitTriggered = true
                        DungeonStatus:Set("Trạng thái Auto Exit: Boss đã gục, chuẩn bị click thoát...")
                        task.spawn(function() 
                            if WebhookModule and WebhookModule.SendNotification then 
                                WebhookModule.SendNotification("**" .. LocalPlayer.Name .. "** vừa hoàn thành vòng Dungeon 20/20!") 
                            end 
                        end)
                        task.spawn(SafeExitRoutine)
                    end
                else 
                    stage20ClockStarted = false; autoExitTriggered = false 
                end
            end

            -- [3] LOGIC AUTO JOIN OTHERS (Giữ nguyên luồng xử lý sảnh chờ của bạn)
            if autoJoinOthersEnabled and (not labStage or labStage.Text == "Unknown" or labStage.Text == "") then
                pcall(function()
                    local wArea = Workspace:FindFirstChild("Area")
                    local center = wArea and wArea:FindFirstChild("center")
                    local innerArea = center and center:FindFirstChild("Area")
                    local abyssFolder = innerArea and innerArea:FindFirstChild("Abyss")
                    
                    if abyssFolder then
                        local alreadyInSomeoneRoom = false
                        local shouldJumpOut = false
                        local jumpOutTarget = nil
                        local roomToJoin = nil
                        
                        for _, room in ipairs(abyssFolder:GetChildren()) do
                            if string.match(room.Name, "Abyss_") then
                                local platform = room:FindFirstChild("Platform")
                                local board = platform and platform:FindFirstChild("Board")
                                local surfaceGui = board and board:FindFirstChild("SurfaceGui")
                                
                                if surfaceGui and surfaceGui:FindFirstChild("FmEmpty") and surfaceGui:FindFirstChild("FmPrepare") and surfaceGui:FindFirstChild("FmInfo") and surfaceGui:FindFirstChild("FmPlayer") then
                                    if (not surfaceGui.FmEmpty.Visible) and (not surfaceGui.FmPrepare.Visible) and surfaceGui.FmInfo.Visible and surfaceGui.FmPlayer.Visible then
                                        local isOtherPlayer = false
                                        local isMyRoom = false
                                      
                                        for _, desc in ipairs(surfaceGui.FmPlayer:GetDescendants()) do
                                            if desc:IsA("ImageLabel") and string.find(desc.Image, "AvatarHeadShot") then
                                                local extractedId = string.match(desc.Image, "id=(%d+)")
                                                if extractedId then 
                                                    if tonumber(extractedId) ~= myUserId then isOtherPlayer = true else isMyRoom = true end 
                                                    break 
                                                end
                                            end
                                        end
                                   
                                        local targetPos = (room:IsA("Model") and room:GetPivot().Position) or (room:IsA("BasePart") and room.Position)
                                        if targetPos then
                                            local dist = (hrp.Position - targetPos).Magnitude
                                            if isMyRoom and dist < 25 then
                                                shouldJumpOut = true
                                                local leaveAth = room:FindFirstChild("LeaveAth")
                                                if leaveAth and leaveAth:IsA("Attachment") then jumpOutTarget = leaveAth.WorldPosition end
                                            elseif isOtherPlayer then
                                                if dist < 15 then alreadyInSomeoneRoom = true elseif not roomToJoin then roomToJoin = targetPos + Vector3.new(0, 5, 0) end
                                            end
                                        end
                                    end
                                end
                            end
                        end
                        
                        if shouldJumpOut then
                            if jumpOutTarget then hrp.CFrame = CFrame.new(jumpOutTarget + Vector3.new(0, 3, 0)) else hrp.CFrame = hrp.CFrame * CFrame.new(0, 0, 20) end
                            task.wait(2)
                        elseif roomToJoin and not alreadyInSomeoneRoom then
                            hrp.CFrame = CFrame.new(roomToJoin)
                            task.wait(3)
                        end
                    end
                end)
            end
        end
    end)
end
