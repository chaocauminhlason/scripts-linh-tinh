-- ====================================================================
-- MODULE: DUNGEON AUTO (AUTO ATTACK, AUTO EXIT, AUTO JOIN & AUTO CREATE)
-- BẢN CẬP NHẬT: FIX TỌA ĐỘ CAO ĐỘ (Y-OFFSET) ĐỂ ĐÁP XUỐNG MƯỢT HƠN
-- ====================================================================
return function(Window, Utils, WebhookModule)
    local Players = game:GetService("Players")
    local Workspace = game:GetService("Workspace")
    local ReplicatedStorage = game:GetService("ReplicatedStorage")
    local VirtualInputManager = game:GetService("VirtualInputManager")
    
    local LocalPlayer = Players.LocalPlayer
    local playerGui = LocalPlayer:WaitForChild("PlayerGui")
    local myUserId = LocalPlayer.UserId

    -- Biến trạng thái hệ thống
    local masterSwitchEnabled = false
    local autoExitEnabled = false
    local autoExitTriggered = false
    local stage20ClockStarted = false
    local autoAttackEnabled = false
    local autoJoinOthersEnabled = false
    local autoCreateEnabled = false
    local targetPlayerCount = 1
    local targetDifficulty = 1 -- 1: Normal, 2: Hard, 3: Nightmare
    
    local dungeonStuckTimer = 0
    local dungeonLastPosition = nil

    -- API Game
    local MonsterSystem = nil
    pcall(function() MonsterSystem = require(ReplicatedStorage.CommonLogic.Monster.MonsterSystem) end)
    local AbyssSystem = nil
    pcall(function() AbyssSystem = require(ReplicatedStorage.CommonLogic.Abyss.AbyssSystem) end)

    -- Hàm Ép Xuống Thú
    local function SmartDismount()
        local isRiding = LocalPlayer:GetAttribute("RidePetId") ~= nil
        if isRiding then
            VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.M, false, game) 
            task.wait() 
            VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.M, false, game)
            task.wait(0.3)
            local char = LocalPlayer.Character
            local humanoid = char and char:FindFirstChild("Humanoid")
            if humanoid then humanoid.Jump = true end
            task.wait(0.1)
        end
    end

    -- ==========================================
    -- GIAO DIỆN UI (RAYFIELD TABS)
    -- ==========================================
    local DungeonTab = Window:CreateTab("Hầm Ngục", 4483362458)
    local DungeonStatus = DungeonTab:CreateLabel("AUTO DUNGEON ĐANG TẮT")

    DungeonTab:CreateToggle({
        Name = "Công Tắc Tổng (Master Switch)",
        CurrentValue = false,
        Flag = "DungeonMasterBreaker",
        Callback = function(Value)
            masterSwitchEnabled = Value
            if not Value then
                DungeonStatus:Set("AUTO DUNGEON ĐANG TẮT")
                pcall(function()
                    local char = LocalPlayer.Character
                    if char and char:FindFirstChild("Humanoid") then
                        char.Humanoid:MoveTo(char.HumanoidRootPart.Position)
                    end
                end)
            else
                DungeonStatus:Set("AUTO DUNGEON ĐANG BẬT")
            end
        end
    })

    DungeonTab:CreateSection("--- LOGIC TRONG HẦM NGỤC ---")
    DungeonTab:CreateToggle({ Name = "Tự Động Đánh Quái (Tối Ưu Phím M)", CurrentValue = false, Flag = "AutoAttack_V2", Callback = function(Value) autoAttackEnabled = Value end })
    DungeonTab:CreateToggle({ Name = "Tự Động Thoát Màn 20", CurrentValue = false, Flag = "AutoExitDungeon", Callback = function(Value) autoExitEnabled = Value if not Value then autoExitTriggered = false stage20ClockStarted = false end end })

    DungeonTab:CreateSection("--- LOGIC NGOÀI SẢNH (LOBBY) ---")
    DungeonTab:CreateToggle({ 
        Name = "Tự Động Join Phòng Người Khác (Ưu tiên số 1)", 
        CurrentValue = false, 
        Flag = "AutoJoinOthers", 
        Callback = function(Value) autoJoinOthersEnabled = Value end 
    })
    
    DungeonTab:CreateToggle({ 
        Name = "Tự Động Tạo Phòng & Start (Bị tắt nếu đang Join)", 
        CurrentValue = false, 
        Flag = "AutoCreateRoom", 
        Callback = function(Value) autoCreateEnabled = Value end 
    })

    DungeonTab:CreateDropdown({
        Name = "Chọn Độ Khó (Khi Tạo Phòng)",
        Options = {"1 (Normal)", "2 (Hard)", "3 (Nightmare)"},
        CurrentOption = {"1 (Normal)"},
        MultipleOptions = false,
        Flag = "DungeonDifficulty",
        Callback = function(Option)
            if Option[1] == "1 (Normal)" then targetDifficulty = 1
            elseif Option[1] == "2 (Hard)" then targetDifficulty = 2
            elseif Option[1] == "3 (Nightmare)" then targetDifficulty = 3
            end
        end,
    })

    DungeonTab:CreateSlider({
        Name = "Số người chờ để Start (1 = Solo)",
        Range = {1, 4},
        Increment = 1,
        Suffix = "Người",
        CurrentValue = 1,
        Flag = "TargetPlayerCount",
        Callback = function(Value) targetPlayerCount = Value end,
    })

    -- LOGIC QUÉT UI THOÁT
    local function SafeExitRoutine()
        local exitBtn = Utils.FindUIElementByName(playerGui, "BgLeave")
        if exitBtn and exitBtn.AbsoluteSize.X > 0 then Utils.ClickButtonExact(exitBtn, "Nút Thoát") else autoExitTriggered = false return end
        
        task.wait(1.5)
        local mainGui = playerGui:FindFirstChild("MainGui")
        if mainGui and mainGui:FindFirstChild("ScreenGui") and mainGui.ScreenGui:FindFirstChild("ConfirmView") then
            local fmBottom = mainGui.ScreenGui.ConfirmView:FindFirstChild("FmBottom")
            if fmBottom and fmBottom:FindFirstChild("BtOk") and fmBottom.BtOk.AbsoluteSize.X > 0 then
                for i = 1, 3 do Utils.ClickButtonExact(fmBottom.BtOk, "Nút Yes") task.wait(0.2) end
            else autoExitTriggered = false return end
        else autoExitTriggered = false return end
        
        task.wait(2.5)
        local checkLabStage = Utils.FindUIElementByName(playerGui, "LabStage")
        local currentStageText = checkLabStage and checkLabStage.Text or "Unknown"
        if currentStageText == "Unknown" or not string.match(currentStageText, "20/20") then 
            autoExitTriggered = false; stage20ClockStarted = false 
        else autoExitTriggered = false end
    end

    -- ==========================================
    -- VÒNG LẶP CORE 
    -- ==========================================
    task.spawn(function()
        while task.wait(0.5) do
            if not masterSwitchEnabled then continue end

            local char = LocalPlayer.Character
            local hrp = char and char:FindFirstChild("HumanoidRootPart")
            local humanoid = char and char:FindFirstChild("Humanoid")
           
            if not (hrp and humanoid and humanoid.Health > 0) then continue end
            local labStage = Utils.FindUIElementByName(playerGui, "LabStage")
            local isInDungeon = (labStage and labStage.Text ~= "Unknown" and labStage.Text ~= "")
         
            -- [1] LOGIC AUTO ATTACK & AUTO EXIT TRONG HẦM NGỤC
            if isInDungeon then
                if autoAttackEnabled then
                    local monstersFolder = Workspace:FindFirstChild("Monsters")
                    local closestDist2D = math.huge
                    local bestTarget = nil

                    if monstersFolder then
                        for _, monster in pairs(monstersFolder:GetChildren()) do
                            if monster:IsA("Model") or monster:IsA("BasePart") then
                                local hum = monster:FindFirstChildOfClass("Humanoid")
                                if (not hum) or (hum and hum.Health > 0) then
                                    local pos = monster:IsA("Model") and monster.PrimaryPart and monster.PrimaryPart.Position or monster.Position
                                    local dist2D = math.sqrt((hrp.Position.X - pos.X)^2 + (hrp.Position.Z - pos.Z)^2)
                                    
                                    if dist2D < closestDist2D then 
                                        closestDist2D = dist2D
                                        bestTarget = monster 
                                    end
                                end
                            end
                        end
                    end

                    if bestTarget then
                        local targetPos = bestTarget:IsA("Model") and bestTarget.PrimaryPart and bestTarget.PrimaryPart.Position or bestTarget.Position
                        
                        if closestDist2D > 40 then
                            hrp.CFrame = CFrame.new(targetPos + Vector3.new(0, 0, 4)) 
                            task.wait(0.2)
                            humanoid:MoveTo(hrp.Position) 
                            dungeonLastPosition = hrp.Position 
                            dungeonStuckTimer = 0
                            SmartDismount()
                        elseif closestDist2D > 12 then
                            humanoid:MoveTo(targetPos)
                            if dungeonLastPosition and (hrp.Position - dungeonLastPosition).Magnitude < 1 then
                                dungeonStuckTimer = dungeonStuckTimer + 0.5
                                if dungeonStuckTimer >= 3 then hrp.CFrame = CFrame.new(targetPos + Vector3.new(0, 0, 3)) dungeonStuckTimer = 0 end
                            else dungeonStuckTimer = 0 end
                            dungeonLastPosition = hrp.Position
                        else
                            humanoid:MoveTo(hrp.Position)
                            dungeonStuckTimer = 0
                            SmartDismount()
                            
                            local id = tonumber(bestTarget.Name:match("Monster_(%d+)"))
                            if id and MonsterSystem then MonsterSystem.ClientAttackMonster(id) end
                        end
                    else 
                        dungeonStuckTimer = 0 
                        SmartDismount()
                    end
                end

                if autoExitEnabled and not autoExitTriggered then
                    local fmTimeFrame = nil
                    pcall(function() fmTimeFrame = playerGui.MainGui.ScreenGui.ArenaMainRightTopView.FmTime end)
                    local currentStage, maxStage = string.match(labStage.Text, "(%d+)/(%d+)")
                    
                    if currentStage and maxStage and tonumber(currentStage) == tonumber(maxStage) then
                        if fmTimeFrame and fmTimeFrame.Visible == true then
                            if not stage20ClockStarted then stage20ClockStarted = true DungeonStatus:Set("Trạng thái: Đồng hồ màn 20 đã chạy...") end
                        end
                        if stage20ClockStarted and (not fmTimeFrame or fmTimeFrame.Visible == false) then
                            autoExitTriggered = true
                            DungeonStatus:Set("Trạng thái: Boss gục, chuẩn bị click thoát...")
                            task.spawn(function() if WebhookModule and WebhookModule.SendNotification then WebhookModule.SendNotification("**" .. LocalPlayer.Name .. "** vừa hoàn thành vòng Dungeon 20/20!") end end)
                            task.spawn(SafeExitRoutine)
                        end
                    else stage20ClockStarted = false; autoExitTriggered = false end
                end

            -- [3] LOGIC LOBBY: TẠO PHÒNG, CHỌN ĐỘ KHÓ, BẤM START & JOIN
            else
                if autoJoinOthersEnabled or autoCreateEnabled then
                    pcall(function()
                        local wArea = Workspace:FindFirstChild("Area")
                        local center = wArea and wArea:FindFirstChild("center")
                        local innerArea = center and center:FindFirstChild("Area")
                        local abyssFolder = innerArea and innerArea:FindFirstChild("Abyss")
                        
                        if abyssFolder then
                            local alreadyInSomeoneRoom = false
                            local isMyRoom = false
                            local myRoomNode = nil
                            local shouldJumpOut = false
                            local jumpOutTarget = nil
                            local roomToJoin = nil
                            local emptyRoomToCreate = nil
                            
                            for _, room in ipairs(abyssFolder:GetChildren()) do
                                if string.match(room.Name, "Abyss_") then
                                    local platform = room:FindFirstChild("Platform")
                                    local board = platform and platform:FindFirstChild("Board")
                                    local surfaceGui = board and board:FindFirstChild("SurfaceGui")
                                    
                                    -- Check xem mình có phải chủ phòng không qua Attribute Server
                                    for k, _ in pairs(room:GetAttributes()) do
                                        if k == "Owner_" .. tostring(myUserId) then
                                            isMyRoom = true
                                            myRoomNode = room
                                            break
                                        end
                                    end

                                    if surfaceGui and surfaceGui:FindFirstChild("FmEmpty") and surfaceGui:FindFirstChild("FmPrepare") and surfaceGui:FindFirstChild("FmInfo") and surfaceGui:FindFirstChild("FmPlayer") then
                                        
                                        -- Lọc phòng có người để Join
                                        if (not surfaceGui.FmEmpty.Visible) and (not surfaceGui.FmPrepare.Visible) and surfaceGui.FmInfo.Visible and surfaceGui.FmPlayer.Visible then
                                            local isOtherPlayer = false
                                            for _, desc in ipairs(surfaceGui.FmPlayer:GetDescendants()) do
                                                if desc:IsA("ImageLabel") and string.find(desc.Image, "AvatarHeadShot") then
                                                    local extractedId = string.match(desc.Image, "id=(%d+)")
                                                    if extractedId and tonumber(extractedId) ~= myUserId then isOtherPlayer = true break end
                                                end
                                            end
                                            
                                            local targetPos = (room:IsA("Model") and room:GetPivot().Position) or (room:IsA("BasePart") and room.Position)
                                            if targetPos then
                                                local dist = (hrp.Position - targetPos).Magnitude
                                                if isMyRoom and dist < 25 then 
                                                    shouldJumpOut = true 
                                                    local leaveAth = room:FindFirstChild("LeaveAth") 
                                                    if leaveAth and leaveAth:IsA("Attachment") then jumpOutTarget = leaveAth.WorldPosition end
                                                elseif isOtherPlayer then 
                                                    if dist < 15 then alreadyInSomeoneRoom = true 
                                                    elseif not roomToJoin then roomToJoin = targetPos + Vector3.new(0, 3, 0) end -- Hạ độ cao Join
                                                end
                                            end
                                        end

                                        -- Lọc phòng trống để Create
                                        if surfaceGui.FmEmpty.Visible and not autoJoinOthersEnabled then
                                            local targetPos = (room:IsA("Model") and room:GetPivot().Position) or (room:IsA("BasePart") and room.Position)
                                            if targetPos and not emptyRoomToCreate then
                                                emptyRoomToCreate = targetPos + Vector3.new(0, 3, 0) -- Hạ độ cao Create
                                            end
                                        end
                                    end
                                end
                            end

                            -- THỰC THI
                            if autoJoinOthersEnabled then
                                DungeonStatus:Set("Sảnh: Đang tìm phòng có người...")
                                if shouldJumpOut then 
                                    if jumpOutTarget then hrp.CFrame = CFrame.new(jumpOutTarget + Vector3.new(0, 3, 0)) else hrp.CFrame = hrp.CFrame * CFrame.new(0, 0, 20) end task.wait(2)
                                elseif roomToJoin and not alreadyInSomeoneRoom then 
                                    hrp.CFrame = CFrame.new(roomToJoin) task.wait(3) 
                                end

                            elseif autoCreateEnabled then
                                if isMyRoom and myRoomNode then
                                    -- ĐÃ TẠO XONG -> Đếm người qua Attributes -> START
                                    local currentPlayerCount = 0
                                    for key, _ in pairs(myRoomNode:GetAttributes()) do
                                        if string.find(key, "Owner_") == 1 or string.find(key, "Mem_") == 1 then
                                            currentPlayerCount = currentPlayerCount + 1
                                        end
                                    end
                                    
                                    if currentPlayerCount >= targetPlayerCount then
                                        DungeonStatus:Set("Sảnh: Đủ " .. currentPlayerCount .. "/" .. targetPlayerCount .. " người! Đang Start...")
                                        
                                        -- Lớp 1: Gọi API Gốc qua bộ Framework của Game (Chuẩn nhất)
                                        pcall(function()
                                            if _G.PathTool and _G.PathTool.AbyssSystem and _G.PathTool.ViewUtil then
                                                _G.PathTool.ViewUtil.DoRequest(_G.PathTool.AbyssSystem.ClientStartAbyss)
                                            end
                                        end)
                                        
                                        -- Lớp 2: Gọi Require gốc
                                        if AbyssSystem and AbyssSystem.ClientStartAbyss then
                                            pcall(function() AbyssSystem.ClientStartAbyss() end)
                                        end
                                        
                                        -- Lớp 3: Quét Click UI thủ công
                                        local startNames = {"BtStart", "BtnStart", "BtGo", "BtEnter"}
                                        for _, btnName in ipairs(startNames) do
                                            local btn = Utils.FindUIElementByName(playerGui, btnName)
                                            if btn and btn.AbsoluteSize.X > 0 and btn.Visible then
                                                Utils.ClickButtonExact(btn, "Nút Start")
                                                break
                                            end
                                        end
                                        
                                        task.wait(2.5) -- Đợi server load map
                                    else
                                        DungeonStatus:Set("Sảnh: Đang chờ người " .. currentPlayerCount .. "/" .. targetPlayerCount)
                                    end

                                elseif emptyRoomToCreate then
                                    -- CHƯA CÓ PHÒNG -> Nhảy vào phòng trống & Setup
                                    local distToEmpty = (hrp.Position - emptyRoomToCreate).Magnitude
                                    if distToEmpty > 10 then
                                        DungeonStatus:Set("Sảnh: Đang nhảy vào phòng trống...")
                                        hrp.CFrame = CFrame.new(emptyRoomToCreate)
                                        task.wait(2)
                                    else
                                        DungeonStatus:Set("Sảnh: Đang thiết lập Độ Khó và Tạo...")
                                        task.wait(1.5) -- Chờ UI Game nảy lên ổn định
                                        
                                        local diffBtnName = tostring(targetDifficulty) .. "Btn"
                                        local diffBtn = nil
                                        local btCreate = nil
                                        
                                        -- Quét kĩ bên trong UI Tạo phòng để tránh bấm nhầm
                                        for _, v in pairs(playerGui:GetDescendants()) do
                                            if v:IsA("GuiObject") then
                                                if v.Name == diffBtnName and v.Parent and v.Parent.Name == "BtnFrame" then
                                                    diffBtn = v
                                                elseif v.Name == "BtCreate" and v.Parent and v.Parent.Name == "BottomFrame" then
                                                    btCreate = v
                                                end
                                            end
                                        end
                                        
                                        -- 1. Chọn độ khó
                                        if diffBtn then
                                            Utils.ClickButtonExact(diffBtn, "Nút Độ Khó " .. targetDifficulty)
                                            task.wait(0.5)
                                        end
                                        
                                        -- 2. Click Tạo phòng
                                        if btCreate then
                                            Utils.ClickButtonExact(btCreate, "Nút Tạo Phòng")
                                            task.wait(2.5) -- Tăng độ trễ cho server xác lập quyền Owner
                                        end
                                    end
                                end
                            end
                        end
                    end)
                else
                    DungeonStatus:Set("Sảnh: Auto Join/Create đang tắt")
                end
            end
        end
    end)
end
