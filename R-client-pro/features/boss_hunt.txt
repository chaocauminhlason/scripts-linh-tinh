-- ====================================================================
-- MODULE: SĂN BOSS VIP (CHỈ QUÉT VÀ TẤN CÔNG TỪ XA)
-- ====================================================================
return function(Window, Utils)
    local Players = game:GetService("Players")
    local ReplicatedStorage = game:GetService("ReplicatedStorage")
    local LocalPlayer = Players.LocalPlayer

    local BossTab = Window:CreateTab("Săn Boss", 4483345998)

    local BOSS_KEYWORDS = {
        "flaragon", "godzilla", "shake", "knight", "tide", "vex", 
        "frost", "wyrm", "gai", "draco", "thunder", "claw", 
        "lampyr", "scare", "reaper", "demon", "lord", "jared"
    }

    -- ==========================================
    -- LOGIC TÌM & EXTRACT ID BOSS
    -- ==========================================
    local function AutoDetectBossID()
        local monstersFolder = workspace:FindFirstChild("Monsters") or workspace:FindFirstChild("Monster")
        local scanTarget = monstersFolder and monstersFolder:GetChildren() or workspace:GetDescendants()

        for _, obj in pairs(scanTarget) do
            if obj:IsA("Model") then
                local hum = obj:FindFirstChildOfClass("Humanoid")
                
                if hum and hum.MaxHealth and hum.MaxHealth >= 100000 and obj.Name ~= LocalPlayer.Name then
                    local nameLower = obj.Name:lower()
                    local isKnownBoss = false
                    
                    for _, k in pairs(BOSS_KEYWORDS) do
                        if nameLower:find(k) then isKnownBoss = true; break end
                    end

                    if isKnownBoss then
                        local bossId = nil
                        local idFromName = string.match(obj.Name, "%d+")
                        if idFromName then bossId = tonumber(idFromName) end
                        
                        if not bossId then
                            bossId = obj:GetAttribute("Id") or obj:GetAttribute("ID") or obj:GetAttribute("Uid") or obj:GetAttribute("TmplId")
                        end
                        
                        if bossId then return bossId, obj.Name end
                    end
                end
            end
        end
        return nil, nil
    end

    -- ==========================================
    -- GIAO DIỆN & LOGIC AUTO TẤN CÔNG
    -- ==========================================
    local attackSettings = { enabled = false, hitSpeed = 0.1 }

    BossTab:CreateSection("--- AUTO ĐÁNH BOSS TỪ XA ---")
    
    local StatusLabel = BossTab:CreateLabel("Trạng thái: Đang nghỉ")
    local TargetLabel = BossTab:CreateLabel("Mục tiêu hiện tại: Trống")

    BossTab:CreateToggle({
        Name = "Bật Auto Tấn Công Boss (Khi Phát Hiện)",
        CurrentValue = false,
        Flag = "AutoAttackBoss",
        Callback = function(Value)
            attackSettings.enabled = Value
            if not Value then StatusLabel:Set("Trạng thái: Đã tắt") end
        end
    })

    task.spawn(function()
        local DataPullFunc = nil
        pcall(function()
            DataPullFunc = ReplicatedStorage:WaitForChild("CommonLibrary"):WaitForChild("Tool"):WaitForChild("RemoteManager"):WaitForChild("Funcs"):WaitForChild("DataPullFunc")
        end)

        while task.wait(attackSettings.hitSpeed) do
            if attackSettings.enabled then
                local currentBossId, currentBossName = AutoDetectBossID()
                
                if currentBossId and DataPullFunc then
                    TargetLabel:Set("Đang khóa mục tiêu: " .. currentBossName)
                    StatusLabel:Set("Trạng thái: ⚔️ Đang nã sát thương (Remote)!")
                    
                    -- Bắn lệnh tấn công trực tiếp
                    pcall(function()
                        DataPullFunc:InvokeServer("MonsterAttackChannel", currentBossId)
                    end)
                else
                    TargetLabel:Set("Mục tiêu hiện tại: Trống")
                    StatusLabel:Set("Trạng thái: 🔍 Đang quét bản đồ tìm Boss...")
                    task.wait(1) -- Quét chậm lại nếu không thấy Boss để đỡ lag
                end
            end
        end
    end)
end
