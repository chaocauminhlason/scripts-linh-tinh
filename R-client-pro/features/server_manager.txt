    local function GetTargetBossKeywords()
        -- Mặc định nếu không đọc được file
        local kws = {"flaragon", "gildron", "shadeknight", "mountusk", "tidevex", "frostwyrm", "dracospike", "thunderclaw", "lampyr", "scareaper"}
        pcall(function()
            if isfile and isfile("RClient_SelectedBossKeywords.json") then
                kws = HttpService:JSONDecode(readfile("RClient_SelectedBossKeywords.json"))
            end
        end)
        return kws
    end

    local function IsBossAliveLocally()
        local targetKeywords = GetTargetBossKeywords()
        if #targetKeywords == 0 then return false end -- Không chọn con nào thì coi như Boss chết hết để nhảy Server
        
        local monstersFolder = workspace:FindFirstChild("Monsters") or workspace:FindFirstChild("Monster")
        local scanTarget = monstersFolder and monstersFolder:GetChildren() or workspace:GetDescendants()
        for _, obj in pairs(scanTarget) do
            if obj:IsA("Model") then
                local hum = obj:FindFirstChildOfClass("Humanoid")
                if hum and hum.MaxHealth and hum.MaxHealth >= 100000 and obj.Name ~= LocalPlayer.Name then
                    local nameLower = obj.Name:lower()
                    for _, k in pairs(targetKeywords) do 
                        if nameLower:find(k) then return true end 
                    end
                end
            end
        end
        return false
    end
