-- ====================================================================
-- MODULE: SERVER MANAGER (V3.0 - REFACTORED & OPTIMIZED)
-- ====================================================================
return function(Window, Utils)
    local Players = game:GetService("Players")
    local ReplicatedStorage = game:GetService("ReplicatedStorage")
    local LocalPlayer = Players.LocalPlayer

    -- Config Settings
    local weatherHopSettings = { enabled = false, targetIds = {60001} }
    local bossHopSettings = { enabled = false }

    _G.ServerManager = {
        isBossPresent = false,
        pauseFarm = false,
        currentWeather = "Bình Thường"
    }

    -- UI Setup
    local ServerTab = Window:CreateTab("Quản Lý Server", 4483362458)
    local WeatherStatus = ServerTab:CreateLabel("Thời tiết hiện tại: Đang tải...")
    
    ServerTab:CreateSection("--- TÌM KIẾM THỜI TIẾT & BOSS ---")

    ServerTab:CreateToggle({
        Name = "Tự Động Nhảy Server Săn Thời Tiết",
        CurrentValue = false,
        Callback = function(Value) weatherHopSettings.enabled = Value end
    })

    ServerTab:CreateToggle({ 
        Name = "Tự Động Nhảy Server Săn Boss", 
        CurrentValue = false, 
        Callback = function(V) bossHopSettings.enabled = V end 
    })

    ServerTab:CreateButton({ 
        Name = "🌐 HOP SERVER TÌM BOSS", 
        Callback = function() Utils.HopServer("Người dùng yêu cầu hop server...") end 
    })

    -- ==========================================
    -- MASTER LOOP: ĐIỀU PHỐI HỆ THỐNG
    -- ==========================================
    task.spawn(function()
        local currentlyInTargetWeather = false

        while task.wait(2) do
            if _G.SystemController and _G.SystemController.shouldStop then continue end

            -- 1. Quét Boss (Dùng Utils)
            local bossKeywords = Utils.LoadJSON("RClient_SelectedBossKeywords.json")
            local bossInfo = Utils.ScanBossByKeywords(bossKeywords)
            
            _G.ServerManager.isBossPresent = (bossInfo ~= nil)

            -- 2. Ưu tiên Boss (Nếu đang săn Boss và thấy boss thì dừng farm)
            if bossInfo and bossHopSettings.enabled then
                _G.ServerManager.pauseFarm = true
                continue 
            end

            -- 3. Logic Thời tiết
            if weatherHopSettings.enabled then
                local WeatherMap = {
                    ["🔴 Huyết Nguyệt (Blood Moon)"] = 60001,
                    ["🐉 Hoàng Hôn Rồng (Draconic Twilight)"] = 60002,
                    ["🌟 Hào Quang Rồng (Dragon-Glow)"] = 60003,
                    ["🧚 Màn Sương Ảo Mộng (Dreamveil)"] = 60004
                }

                local currentName, currentId = Utils.GetCurrentWeather(WeatherMap)
                _G.ServerManager.currentWeather = currentName
                WeatherStatus:Set("Thời tiết: " .. currentName)

                local isTarget = false
                for _, tid in ipairs(weatherHopSettings.targetIds) do
                    if currentId == tid then isTarget = true break end
                end

                if isTarget then
                    _G.ServerManager.pauseFarm = false
                    currentlyInTargetWeather = true
                else
                    _G.ServerManager.pauseFarm = true
                    if currentlyInTargetWeather then
                        game.StarterGui:SetCore("SendNotification", {Title = "Thời tiết đã hết!", Text = "Đang hop server...", Duration = 5})
                        currentlyInTargetWeather = false
                    end
                    Utils.HopServer("Đang tìm thời tiết VIP...")
                end
            else
                _G.ServerManager.pauseFarm = false
            end
        end
    end)
    
    print("✅ Server Manager 3.0 (Refactored) loaded!")
end
