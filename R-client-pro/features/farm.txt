-- ====================================================================
-- MODULE: R-GAME (AUTO BẮT, ĐÁNH, HỒI MÁU - BẢN FULL INTEGRATED V9.6 FINAL)
-- ====================================================================
return function(Window, Utils)
    local Players = game:GetService("Players")
    local Workspace = game:GetService("Workspace")
    local ReplicatedStorage = game:GetService("ReplicatedStorage")
    local HttpService = game:GetService("HttpService")
    local LocalPlayer = Players.LocalPlayer

    -- ==========================================
    -- HOÀN THIỆN CẤU HÌNH STATE & ĐĂNG KÝ HỆ THỐNG
    -- ==========================================
    local camSettings = {
        autoAttack = false,
        autoCatch = false,
        autoHeal = false,
        isHealing = false,
        savedPosition = nil,
        targetMonsterNames = {},
        yieldForBoss = true,
        autoReturnToFarm = false,
        healThreshold = 1,
        attackRadius = 200
        local MonsterDropdown = nil
    }

    if _G.SystemController and type(_G.SystemController.registerModule) == "function" then
        _G.SystemController:registerModule("AutoFarm", function(paused)
            camSettings.autoAttack = not paused
            camSettings.autoCatch = not paused
        end)
    end

    local savedFarmPosition = nil
    local posFileName = "R_ClientPro_FarmPos.json"
    local monsterFilterFile = "R_ClientPro_MonsterFilter.json"

    local MonsterSystem = nil
    pcall(function() MonsterSystem = require(ReplicatedStorage.CommonLogic.Monster.MonsterSystem) end)

    -- ==========================================
    -- HÀM BỔ TRỢ (ĐÃ FIX: UPVALUES & CLIENTMONSTERS)
    -- ==========================================
    local MonsterNameCache = {}
    local InternalMonsterTable = nil

    -- Khởi tạo kết nối vào vùng nhớ ẩn của game
    local function InitMonsterMemory()
        if InternalMonsterTable then return end
        pcall(function()
            local env = getrenv and getrenv()._G and getrenv()._G.PathTool
            if env and env.MgrMonsterClient then
                local upvalues = debug.getupvalues(env.MgrMonsterClient.GetMonsterInfo)
                if upvalues and type(upvalues[1]) == "table" then
                    InternalMonsterTable = upvalues[1]
                end
            end
        end)
    end

    local function GetCachedMonsterName(monster)
        if not monster then return "Unknown" end
        
        -- Bóc UID từ tên ngoài map (VD: "Monster_46808" -> 46808)
        local uidStr = string.match(monster.Name, "Monster_(%d+)")
        if not uidStr then return monster.Name end
        
        local uidNum = tonumber(uidStr)
        if MonsterNameCache[uidNum] then return MonsterNameCache[uidNum] end
        
        InitMonsterMemory()
        
        if InternalMonsterTable and type(InternalMonsterTable) == "table" then
            local rawData = InternalMonsterTable[uidNum]
            if type(rawData) == "table" then
                local tmplId = rawData.TmplId or rawData.tmplId or rawData.cfgId
                if tmplId then
                    local env = getrenv()._G.PathTool
                    if env.CfgMonster and env.CfgMonster.Tmpls then
                        local tmpls = env.CfgMonster.Tmpls
                        local cfg = tmpls[tostring(tmplId)] or tmpls[tonumber(tmplId)]
                        if cfg then
                            local realName = cfg.Name or cfg.name or cfg.Title
                            if realName then
                                MonsterNameCache[uidNum] = realName
                                return realName
                            end
                        end
                    end
                end
            end
        end
        
        return monster.Name
    end

-- Đảm bảo hàm này nằm TẠI ĐÂY, trong cùng 1 file với các hàm khác
local function GetCachedMonsterName(monster)
    -- ... (giữ nguyên logic lấy tên như tôi đã gửi ở trên) ...
    -- Nếu bạn lỡ tay xóa hoặc làm mất, hãy chép lại đúng code từ phần "HÀM BỔ TRỢ"
end

-- Sửa lại hàm quét trong file Auto Farm thành thế này:
local function GetAllMonsterNames()
    local monsterNames = {}
    local nameDict = {}
    local clientMonsters = Workspace:FindFirstChild("ClientMonsters")
    
    if clientMonsters then
        for _, obj in pairs(clientMonsters:GetChildren()) do
            -- Dùng pcall để nếu hàm dịch tên lỗi thì không làm crash cả hệ thống quét
            local success, realName = pcall(function() return GetCachedMonsterName(obj) end)
            
            if success and realName and realName ~= "Unknown" and not nameDict[realName] then
                nameDict[realName] = true
                table.insert(monsterNames, realName)
            end
        end
    end
    return monsterNames
end

    local function IsMonsterInFilter(monster)
        if not camSettings.targetMonsterNames or #camSettings.targetMonsterNames == 0 then return true end
        local realName = GetCachedMonsterName(monster)
        for _, targetName in ipairs(camSettings.targetMonsterNames) do
            if string.find(string.lower(realName), string.lower(targetName)) then return true end
        end
        return false
    end

    local function SafeTeleport(targetCFrame, offsetY)
        offsetY = offsetY or 5
        local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        if not hrp then return false end
        local targetPos = targetCFrame.Position + Vector3.new(0, offsetY, 0)
        local rayOrigin = targetPos + Vector3.new(0, 10, 0)
        local rayDirection = Vector3.new(0, -20, 0)
        local raycastParams = RaycastParams.new()
        raycastParams.FilterType = Enum.RaycastFilterType.Blacklist
        raycastParams.FilterDescendantsInstances = {LocalPlayer.Character}
        local rayResult = workspace:Raycast(rayOrigin, rayDirection, raycastParams)
        if rayResult then targetPos = Vector3.new(targetPos.X, rayResult.Position.Y + 3, targetPos.Z) end
        hrp.CFrame = CFrame.new(targetPos)
        return true
    end

    local function LoadSavedPosition()
        if isfile and isfile(posFileName) and readfile then
            local success, data = pcall(function() return HttpService:JSONDecode(readfile(posFileName)) end)
            if success and data and data.X and data.Y and data.Z then return CFrame.new(data.X, data.Y, data.Z) end
        end
        return nil
    end

    local function LoadMonsterFilter()
        if isfile and isfile(monsterFilterFile) and readfile then
            local success, data = pcall(function() return HttpService:JSONDecode(readfile(monsterFilterFile)) end)
            if success and type(data) == "table" then return data end
        end
        return {}
    end

    savedFarmPosition = LoadSavedPosition()
    camSettings.targetMonsterNames = LoadMonsterFilter()

    -- ==========================================
    -- THIẾT KẾ GIAO DIỆN UI (RAYFIELD)
    -- ==========================================
    local CamTab = Window:CreateTab("Auto farm", 4483345998)

    CamTab:CreateSection("--- ĐIỀU KHIỂN TỔNG ---")
    CamTab:CreateButton({
        Name = "🛑 STOP TẤT CẢ HỆ THỐNG",
        Callback = function()
            if _G.SystemController then
                _G.SystemController.IsPaused = true
                for name, pauseFunc in pairs(_G.SystemController.Modules) do pcall(function() pauseFunc(true) end) end
                game.StarterGui:SetCore("SendNotification", { Title = "Hệ thống", Text = "🛑 Đã DỪNG toàn bộ!", Duration = 3 })
            end
        end
    })
    CamTab:CreateButton({
        Name = "▶️ RESUME TẤT CẢ HỆ THỐNG",
        Callback = function()
            if _G.SystemController then
                _G.SystemController.IsPaused = false
                for name, pauseFunc in pairs(_G.SystemController.Modules) do pcall(function() pauseFunc(false) end) end
                game.StarterGui:SetCore("SendNotification", { Title = "Hệ thống", Text = "▶️ Tiếp tục vận hành!", Duration = 3 })
            end
        end
    })

    CamTab:CreateSection("--- QUẢN LÝ QUÁI FARM ---")
    local MonsterListLabel = CamTab:CreateLabel("🐾 Đang farm: Tất cả quái")
    
-- 1. Đảm bảo UpdateMonsterLabel được định nghĩa trước
    local function UpdateMonsterLabel()
        if MonsterListLabel then
            if #camSettings.targetMonsterNames > 0 then
                MonsterListLabel:Set("🐾 Đang farm: " .. table.concat(camSettings.targetMonsterNames, ", "))
            else
                MonsterListLabel:Set("🐾 Đang farm: Tất cả quái")
            end
        end
    end

    -- 2. Khai báo Dropdown chính xác
    MonsterDropdown = CamTab:CreateDropdown({
        Name = "Chọn quái farm",
        Options = {"Tất cả quái"},
        CurrentOption = {"Tất cả quái"},
        MultipleOptions = true,
        Flag = "MonsterSelectFarm",
        Callback = function(Options)
            pcall(function()
                if table.find(Options, "Tất cả quái") then
                    camSettings.targetMonsterNames = {}
                else
                    camSettings.targetMonsterNames = Options
                end
                
                if writefile then 
                    pcall(function() writefile(monsterFilterFile, HttpService:JSONEncode(camSettings.targetMonsterNames)) end)
                end
                UpdateMonsterLabel()
            end)
        end
    })

    -- 3. Nút Quét với cơ chế Refresh an toàn
    CamTab:CreateButton({
        Name = "🔍 Quét danh sách quái trên Map",
        Callback = function()
            local names = GetAllMonsterNames()
            local finalOptions = {"Tất cả quái"}
            for _, n in pairs(names) do table.insert(finalOptions, n) end
            
            -- Gọi Refresh của Rayfield (Nếu có)
            if MonsterDropdown and MonsterDropdown.Refresh then
                MonsterDropdown:Refresh(finalOptions, true)
            else
                -- Dự phòng nếu không có Refresh
                MonsterDropdown:SetOptions(finalOptions)
            end
            
            game.StarterGui:SetCore("SendNotification", { Title = "✅ Đã quét", Text = "Đã tìm thấy " .. #names .. " quái", Duration = 3 })
        end
    })
    
    -- Cập nhật label lần đầu
    UpdateMonsterLabel()

    CamTab:CreateSection("--- CÀI ĐẶT FARM ---")
    CamTab:CreateToggle({ Name = "Tự Động Đánh Quái", CurrentValue = false, Flag = "FarmAutoAttack", Callback = function(V) camSettings.autoAttack = V end })
    CamTab:CreateSlider({ Name = "Bán kính Farm", Range = {10, 2000}, Increment = 10, Suffix = " studs", CurrentValue = 200, Flag = "FarmRadiusSlider", Callback = function(V) camSettings.attackRadius = V end })
    CamTab:CreateToggle({ Name = "Tự Động Bắt Quái", CurrentValue = false, Flag = "FarmAutoCatch", Callback = function(V) camSettings.autoCatch = V end })
    CamTab:CreateToggle({ Name = "Tạm dừng khi có Boss", CurrentValue = true, Flag = "FarmYieldForBoss", Callback = function(V) camSettings.yieldForBoss = V end })
    
    CamTab:CreateSection("--- TỰ ĐỘNG VỀ FARM ---")
    CamTab:CreateToggle({
        Name = "Tự động về vị trí farm khi vào game", CurrentValue = false, Flag = "AutoReturnToFarm",
        Callback = function(V)
            camSettings.autoReturnToFarm = V
            if V and savedFarmPosition then task.spawn(function() task.wait(1) SafeTeleport(savedFarmPosition, 5) end) end
        end
    })

    CamTab:CreateSection("--- TỌA ĐỘ FARM ---")
    local initText = savedFarmPosition and string.format("📍 %d, %d, %d", math.floor(savedFarmPosition.Position.X), math.floor(savedFarmPosition.Position.Y), math.floor(savedFarmPosition.Position.Z)) or "📍 Chưa cấu hình"
    local FarmPosLabel = CamTab:CreateLabel("Tọa độ hiện tại: " .. initText)

    CamTab:CreateButton({
        Name = "💾 Lưu vị trí đang đứng làm tâm điểm Farm",
        Callback = function()
            local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
            if hrp then
                savedFarmPosition = hrp.CFrame
                if writefile then pcall(function() writefile(posFileName, HttpService:JSONEncode({X = hrp.Position.X, Y = hrp.Position.Y, Z = hrp.Position.Z})) end) end
                FarmPosLabel:Set(string.format("Tọa độ hiện tại: %d, %d, %d", math.floor(hrp.Position.X), math.floor(hrp.Position.Y), math.floor(hrp.Position.Z)))
                game.StarterGui:SetCore("SendNotification", { Title = "💾 Lưu Thành Công", Text = "Vị trí bãi farm đã được ghi nhớ!", Duration = 3 })
            end
        end
    })

    CamTab:CreateButton({
        Name = "🚀 Dịch chuyển nhanh về tọa độ Farm đã lưu",
        Callback = function()
            if savedFarmPosition then
                SafeTeleport(savedFarmPosition, 5)
                game.StarterGui:SetCore("SendNotification", { Title = "🚀 Teleport", Text = "Đã quay trở về tâm bãi farm", Duration = 3 })
            end
        end
    })

    -- ==========================================
    -- HỆ THỐNG CÁC VÒNG LẶP LIÊN TỤC (LOOPS RUNTIME)
    -- ==========================================
    
    -- 1. LUỒNG TỰ ĐỘNG ĐÁNH QUÁI
    task.spawn(function()
        while task.wait(0.2) do
            if _G.SystemController and _G.SystemController.IsPaused then continue end
            if not camSettings.autoAttack or camSettings.isHealing then continue end
            if camSettings.yieldForBoss and _G.BossStatus and _G.BossStatus.bossFound then continue end
            
            local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
            if not hrp then continue end
            
            local centerPos = savedFarmPosition and savedFarmPosition.Position or hrp.Position
            
            local foldersToScan = {"ClientMonsters", "Monsters"}
            for _, folderName in ipairs(foldersToScan) do
                local monstersFolder = Workspace:FindFirstChild(folderName)
                if monstersFolder then
                    for _, monster in pairs(monstersFolder:GetChildren()) do
                        if monster:IsA("Model") or monster:IsA("BasePart") then
                            -- Lấy lõi vật lý (Fix lỗi không có Humanoid)
                            local root = monster.PrimaryPart or monster:FindFirstChild("HumanoidRootPart") or monster:FindFirstChild("Root") or (monster:IsA("BasePart") and monster)
                            
                            if root and root:IsA("BasePart") and monster:GetAttribute("CatchEndTick") == nil then
                                local distance = (root.Position - centerPos).Magnitude
                                if distance <= camSettings.attackRadius and IsMonsterInFilter(monster) then
                                    local idStr = monster.Name:match("Monster_(%d+)")
                                    if idStr and MonsterSystem then 
                                        pcall(function() MonsterSystem:ClientAttackMonster(tonumber(idStr)) end)
                                    end
                                end
                            end
                        end
                    end
                end
            end
        end
    end)

    -- 2. LUỒNG TỰ ĐỘNG BẮT QUÁI
    task.spawn(function()
        while task.wait(0.5) do
            if _G.SystemController and _G.SystemController.IsPaused then continue end
            if not camSettings.autoCatch or camSettings.isHealing then continue end
            if camSettings.yieldForBoss and _G.BossStatus and _G.BossStatus.bossFound then continue end
            
            local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
            if not hrp then continue end

            local centerPos = savedFarmPosition and savedFarmPosition.Position or hrp.Position
            local playerId = LocalPlayer.UserId
            local targetCatch, closestDist = nil, math.huge

            local foldersToScan = {"ClientMonsters", "Monsters"}
            for _, folderName in ipairs(foldersToScan) do
                local monstersFolder = Workspace:FindFirstChild(folderName)
                if monstersFolder then
                    for _, monster in pairs(monstersFolder:GetChildren()) do
                        if (monster:IsA("Model") or monster:IsA("BasePart")) and IsMonsterInFilter(monster) then
                            local catchTaken = monster:GetAttribute("CatchTakenPlayerId")
                            if catchTaken and catchTaken == playerId then
                                local root = monster.PrimaryPart or monster:FindFirstChild("HumanoidRootPart") or monster:FindFirstChild("Root") or (monster:IsA("BasePart") and monster)
                                if root and root:IsA("BasePart") then
                                    local distFromCenter = (root.Position - centerPos).Magnitude
                                    if distFromCenter <= camSettings.attackRadius then
                                        local distFromPlayer = (hrp.Position - root.Position).Magnitude
                                        if distFromPlayer < closestDist then 
                                            closestDist = distFromPlayer
                                            targetCatch = monster 
                                        end
                                    end
                                end
                            end
                        end
                    end
                end
            end

            if targetCatch and MonsterSystem then
                local targetId = tonumber(targetCatch.Name:match("Monster_(%d+)"))
                local root = targetCatch.PrimaryPart or targetCatch:FindFirstChild("HumanoidRootPart") or targetCatch:FindFirstChild("Root") or (targetCatch:IsA("BasePart") and targetCatch)
                
                if root and root:IsA("BasePart") then SafeTeleport(CFrame.new(root.Position), 5) end
                
                task.spawn(function()
                    local success, isStarted = pcall(function() return MonsterSystem:ClientCatchMonsterStart(targetId) end)
                    if success then
                        if isStarted == false then task.wait(5) else
                            task.wait(1.5)
                            pcall(function() MonsterSystem:ClientCatchMonsterComplete(targetId) end)
                        end
                    end
                end)
            end
        end
    end)

    -- 3. TỰ ĐỘNG QUAY LẠI BÃI FARM KHI VỪA VÀO GAME
    task.spawn(function()
        task.wait(3)
        if camSettings.autoReturnToFarm and savedFarmPosition then
            SafeTeleport(savedFarmPosition, 5)
        end
    end)

    print("✅ Module Farm V9.6 Final (ClientMonsters + Upvalues Logic) Loaded Successfully!")
end
