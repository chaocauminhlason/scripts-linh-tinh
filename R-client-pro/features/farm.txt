-- ====================================================================
-- MODULE: R-GAME (AUTO BẮT, ĐÁNH, HỒI MÁU - BẢN FULL INTEGRATED V10 FINAL)
-- ====================================================================
return function(Window, Utils)
    local Players = game:GetService("Players")
    local Workspace = game:GetService("Workspace")
    local ReplicatedStorage = game:GetService("ReplicatedStorage")
    local HttpService = game:GetService("HttpService")
    local LocalPlayer = Players.LocalPlayer

    local tpFarmSettings = {
    enabled = false,
    attackDelay = 0.2, -- Tốc độ vung chiêu (tránh bị server kick vì spam)
    distanceOffset = Vector3.new(0, 7, 0) -- Teleport lên trên đầu quái 7 studs để né đòn
}

    -- ==========================================
    -- 1. CẤU HÌNH STATE & ĐĂNG KÝ HỆ THỐNG
    -- ==========================================
    local camSettings = {
        autoAttack = false,
        autoCatch = false,
        autoHeal = false,
        isHealing = false,
        savedPosition = nil,
        targetMonsterNames = {},
        
        autoReturnToFarm = false,
        healThreshold = 1,
        attackRadius = 200
    }

    
    if _G.SystemController and type(_G.SystemController.registerModule) == "function" then
        _G.SystemController:registerModule("AutoFarm", function(paused)
            camSettings.autoAttack = not paused
            camSettings.autoCatch = not paused
        end)
    end

    local savedFarmPosition = nil
    local posFileName = "R_ClientPro_FarmPos.json"
    local monsterFilterFile = "R_ClientPro_MonsterFilter.json"

    local MonsterSystem = nil
    pcall(function() MonsterSystem = require(ReplicatedStorage.CommonLogic.Monster.MonsterSystem) end)

    -- ==========================================
    -- 2. HÀM BỔ TRỢ (UPVALUES & DỊCH TÊN)
    -- ==========================================
    local MonsterNameCache = {}
    local InternalMonsterTable = nil

    local function InitMonsterMemory()
        if InternalMonsterTable then return end
        pcall(function()
            local env = getrenv and getrenv()._G and getrenv()._G.PathTool
            if env and env.MgrMonsterClient then
                local upvalues = debug.getupvalues(env.MgrMonsterClient.GetMonsterInfo)
                if upvalues and type(upvalues[1]) == "table" then
                    InternalMonsterTable = upvalues[1]
                end
            end
        end)
    end

    local function GetCachedMonsterName(monster)
        if not monster then return "Unknown" end
        
        local uidStr = string.match(monster.Name, "Monster_(%d+)")
        if not uidStr then return monster.Name end
        
        local uidNum = tonumber(uidStr)
        if MonsterNameCache[uidNum] then return MonsterNameCache[uidNum] end
        
        InitMonsterMemory()
        
        if InternalMonsterTable and type(InternalMonsterTable) == "table" then
            local rawData = InternalMonsterTable[uidNum]
            if type(rawData) == "table" then
                local tmplId = rawData.TmplId or rawData.tmplId or rawData.cfgId
                if tmplId then
                    local env = getrenv()._G.PathTool
                    if env.CfgMonster and env.CfgMonster.Tmpls then
                        local tmpls = env.CfgMonster.Tmpls
                        local cfg = tmpls[tostring(tmplId)] or tmpls[tonumber(tmplId)]
                        if cfg then
                            local realName = cfg.Name or cfg.name or cfg.Title
                            if realName then
                                MonsterNameCache[uidNum] = realName
                                return realName
                            end
                        end
                    end
                end
            end
        end
        return monster.Name
    end

    local function GetAllMonsterNames()
        local monsterNames = {}
        local nameDict = {}
        
        local foldersToScan = {"ClientMonsters", "Monsters"}
        for _, folderName in ipairs(foldersToScan) do
            local folder = Workspace:FindFirstChild(folderName)
            if folder then
                for _, obj in pairs(folder:GetChildren()) do
                    if obj:IsA("Model") or obj:IsA("BasePart") then
                        local success, realName = pcall(function() return GetCachedMonsterName(obj) end)
                        if success and realName and realName ~= "Unknown" and not string.find(realName, "Monster_") and not nameDict[realName] then
                            nameDict[realName] = true
                            table.insert(monsterNames, realName)
                        end
                    end
                end
            end
        end
        
        table.sort(monsterNames)
        return monsterNames
    end

    local function IsMonsterInFilter(monster)
        if not camSettings.targetMonsterNames or #camSettings.targetMonsterNames == 0 then return true end
        local realName = GetCachedMonsterName(monster)
        for _, targetName in ipairs(camSettings.targetMonsterNames) do
            if string.find(string.lower(realName), string.lower(targetName)) then return true end
        end
        return false
    end

    local function SafeTeleport(targetCFrame, offsetY)
        offsetY = offsetY or 5
        local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        if not hrp then return false end
        local targetPos = targetCFrame.Position + Vector3.new(0, offsetY, 0)
        local rayOrigin = targetPos + Vector3.new(0, 20, 0)
        local rayDirection = Vector3.new(0, -20, 0)
        local raycastParams = RaycastParams.new()
        raycastParams.FilterType = Enum.RaycastFilterType.Blacklist
        raycastParams.FilterDescendantsInstances = {LocalPlayer.Character}
        local rayResult = workspace:Raycast(rayOrigin, rayDirection, raycastParams)
        if rayResult then targetPos = Vector3.new(targetPos.X, rayResult.Position.Y + 3, targetPos.Z) end
        hrp.CFrame = CFrame.new(targetPos)
        return true
    end

    local function LoadSavedPosition()
        if isfile and isfile(posFileName) and readfile then
            local success, data = pcall(function() return HttpService:JSONDecode(readfile(posFileName)) end)
            if success and data and data.X and data.Y and data.Z then return CFrame.new(data.X, data.Y, data.Z) end
        end
        return nil
    end

    local function LoadMonsterFilter()
        if isfile and isfile(monsterFilterFile) and readfile then
            local success, data = pcall(function() return HttpService:JSONDecode(readfile(monsterFilterFile)) end)
            if success and type(data) == "table" then return data end
        end
        return {}
    end

    savedFarmPosition = LoadSavedPosition()
    camSettings.targetMonsterNames = LoadMonsterFilter()

    -- ==========================================
    -- 3. THIẾT KẾ GIAO DIỆN UI (RAYFIELD)
    -- ==========================================
    local CamTab = Window:CreateTab("Auto farm", 4483345998)
    local MonsterDropdown = nil -- Khai báo biến toàn cục cho Module

    CamTab:CreateSection("ĐIỀU KHIỂN TỔNG")
    CamTab:CreateButton({
        Name = "🛑 STOP TẤT CẢ",
        Callback = function()
            if _G.SystemController then
                _G.SystemController.IsPaused = true
                for name, pauseFunc in pairs(_G.SystemController.Modules) do pcall(function() pauseFunc(true) end) end
                game.StarterGui:SetCore("SendNotification", { Title = "Hệ thống", Text = "🛑 Đã DỪNG toàn bộ!", Duration = 3 })
            end
        end
    })
    CamTab:CreateButton({
        Name = "▶️ RESUME TẤT CẢ",
        Callback = function()
            if _G.SystemController then
                _G.SystemController.IsPaused = false
                for name, pauseFunc in pairs(_G.SystemController.Modules) do pcall(function() pauseFunc(false) end) end
                game.StarterGui:SetCore("SendNotification", { Title = "Hệ thống", Text = "▶️ Tiếp tục vận hành!", Duration = 3 })
            end
        end
    })

    --CamTab:CreateSection("QUẢN LÝ QUÁI FARM")
    local MonsterListLabel = CamTab:CreateLabel("🐾 Đang farm: Tất cả quái")
    
    local function UpdateMonsterLabel()
        if #camSettings.targetMonsterNames > 0 then
            MonsterListLabel:Set("🐾 Đang farm: " .. table.concat(camSettings.targetMonsterNames, ", "))
        else
            MonsterListLabel:Set("🐾 Đang farm: Tất cả quái")
        end
    end
    UpdateMonsterLabel()

    -- Khởi tạo Dropdown hỗ trợ Multi-select
    local initOptions = #camSettings.targetMonsterNames > 0 and camSettings.targetMonsterNames or {"Tất cả quái"}
    MonsterDropdown = CamTab:CreateDropdown({
        Name = "Chọn quái farm (Multi-select)",
        Options = {"Tất cả quái"},
        CurrentOption = initOptions,
        MultipleOptions = true,
        Flag = "MonsterSelectFarm",
        Callback = function(Options)
            pcall(function()
                if table.find(Options, "Tất cả quái") then
                    camSettings.targetMonsterNames = {}
                else
                    camSettings.targetMonsterNames = Options
                end
                if writefile then pcall(function() writefile(monsterFilterFile, HttpService:JSONEncode(camSettings.targetMonsterNames)) end) end
                UpdateMonsterLabel()
            end)
        end
    })

    -- Nút Quét dùng hàm Refresh
    CamTab:CreateButton({
        Name = "🔍 Quét danh sách quái trên Map",
        Callback = function()
            pcall(function()
                local names = GetAllMonsterNames()
                if #names == 0 then
                    game.StarterGui:SetCore("SendNotification", { Title = "⚠️ Cảnh báo", Text = "Không tìm thấy quái nào!", Duration = 3 })
                    return
                end

                local finalOptions = {"Tất cả quái"}
                for _, name in ipairs(names) do table.insert(finalOptions, name) end
                
                if MonsterDropdown then
                    if MonsterDropdown.Refresh then
                        MonsterDropdown:Refresh(finalOptions, true)
                    else
                        MonsterDropdown:SetOptions(finalOptions)
                    end
                end
                
                game.StarterGui:SetCore("SendNotification", { Title = "✅ Quét Hoàn Tất", Text = "Đã tìm thấy " .. #names .. " loại quái", Duration = 3 })
            end)
        end
    })

    CamTab:CreateSection("FARM")
    CamTab:CreateToggle({ Name = "Tự Động Đánh Quái", CurrentValue = false, Flag = "FarmAutoAttack", Callback = function(V) camSettings.autoAttack = V end })
    CamTab:CreateToggle({
    Name = "Tấn công dịch chuyển", 
    CurrentValue = false, 
    Flag = "AutoFarmTeleportToggle",
    Callback = function(Value)
        tpFarmSettings.enabled = Value
        if Value then
            game.StarterGui:SetCore("SendNotification", { Title = "🚀 Teleport Farm", Text = "Đã kích hoạt chế độ diệt quái không gian!", Duration = 3 })
        end
    end
})
    CamTab:CreateSlider({ Name = "Bán kính Farm", Range = {10, 2000}, Increment = 10, Suffix = " studs", CurrentValue = 200, Flag = "FarmRadiusSlider", Callback = function(V) camSettings.attackRadius = V end })
    CamTab:CreateToggle({ Name = "Tự Động Bắt Quái", CurrentValue = false, Flag = "FarmAutoCatch", Callback = function(V) camSettings.autoCatch = V end })

    CamTab:CreateSection("HỒI MÁU")
    CamTab:CreateToggle({ 
        Name = "Tự Động Hồi Máu Pet", CurrentValue = false, Flag = "FarmAutoHeal", 
        Callback = function(V) pcall(function() camSettings.autoHeal = V end) end 
    })
    CamTab:CreateSlider({
        Name = "Max pet heal", Range = {1, 3}, Increment = 1, Suffix = " Pet", CurrentValue = 1, Flag = "HealThresholdSlider",
        Callback = function(Value) pcall(function() camSettings.healThreshold = Value end) end,
    })

    CamTab:CreateSection("TỰ ĐỘNG VỀ FARM")
    CamTab:CreateToggle({
        Name = "Tự động về vị trí farm khi vào game", CurrentValue = false, Flag = "AutoReturnToFarm",
        Callback = function(V)
            camSettings.autoReturnToFarm = V
            if V and savedFarmPosition then task.spawn(function() task.wait(1) SafeTeleport(savedFarmPosition, 5) end) end
        end
    })

    CamTab:CreateSection("TỌA ĐỘ FARM")
    local initText = savedFarmPosition and string.format("📍 %d, %d, %d", math.floor(savedFarmPosition.Position.X), math.floor(savedFarmPosition.Position.Y), math.floor(savedFarmPosition.Position.Z)) or "📍 Chưa cấu hình"
    local FarmPosLabel = CamTab:CreateLabel("Tọa độ hiện tại: " .. initText)

    CamTab:CreateButton({
        Name = "💾 Lưu vị trí Farm",
        Callback = function()
            local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
            if hrp then
                savedFarmPosition = hrp.CFrame
                if writefile then pcall(function() writefile(posFileName, HttpService:JSONEncode({X = hrp.Position.X, Y = hrp.Position.Y, Z = hrp.Position.Z})) end) end
                FarmPosLabel:Set(string.format("Tọa độ hiện tại: %d, %d, %d", math.floor(hrp.Position.X), math.floor(hrp.Position.Y), math.floor(hrp.Position.Z)))
                game.StarterGui:SetCore("SendNotification", { Title = "💾 Lưu Thành Công", Text = "Vị trí bãi farm đã được ghi nhớ!", Duration = 3 })
            end
        end
    })

    CamTab:CreateButton({
        Name = "🚀 Dịch chuyển về tọa độ Farm",
        Callback = function()
            if savedFarmPosition then
                SafeTeleport(savedFarmPosition, 5)
                game.StarterGui:SetCore("SendNotification", { Title = "🚀 Teleport", Text = "Đã quay trở về tâm bãi farm", Duration = 3 })
            end
        end
    })

    -- ==========================================
    -- 4. HỆ THỐNG CÁC VÒNG LẶP LIÊN TỤC (LOOPS)
    -- ==========================================
    
-- ==========================================
    -- 1. LUỒNG TỰ ĐỘNG ĐÁNH QUÁI (ĐÁNH THƯỜNG)
    -- ==========================================
    task.spawn(function()
        while task.wait(0.2) do
            local controller = _G.SystemController
            if controller and (controller.IsPaused or controller.CurrentLock ~= nil) then continue end
            if _G.PendingBossHunt or _G.PendingSpecialBoss then continue end
            if tpFarmSettings.enabled or camSettings.isHealing or not camSettings.autoAttack then continue end

            local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
            if not hrp then continue end
            
            local centerPos = savedFarmPosition and savedFarmPosition.Position or hrp.Position
            
            -- Gọi 1 dòng duy nhất là ra quái
            local validMonsters = Utils.SmartScanMonsters(centerPos, camSettings.attackRadius, "Attack")

            for _, mData in ipairs(validMonsters) do
                if mData.Id and MonsterSystem then 
                    pcall(function() MonsterSystem:ClientAttackMonster(mData.Id) end)
                end
            end
        end
    end)

    -- ==========================================
    -- 2. LUỒNG TỰ ĐỘNG BẮT QUÁI
    -- ==========================================
    task.spawn(function()
        while task.wait(0.5) do
            local controller = _G.SystemController
            if controller and (controller.IsPaused or controller.CurrentLock ~= nil) then continue end
            if _G.PendingBossHunt or _G.PendingSpecialBoss then continue end
            if not camSettings.autoCatch or camSettings.isHealing then continue end
            
            local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
            if not hrp then continue end

            local centerPos = savedFarmPosition and savedFarmPosition.Position or hrp.Position
            
            -- Quét với mode "Catch"
            local catchableMonsters = Utils.SmartScanMonsters(centerPos, camSettings.attackRadius, "Catch")
            
            -- Tìm con gần nhất
            local targetCatch = nil
            local minDist = math.huge
            for _, mData in ipairs(catchableMonsters) do
                if mData.Distance < minDist then
                    minDist = mData.Distance
                    targetCatch = mData
                end
            end

            if targetCatch and MonsterSystem then
                SafeTeleport(CFrame.new(targetCatch.Position), 5)
                
                task.spawn(function()
                    local success, isStarted = pcall(function() return MonsterSystem:ClientCatchMonsterStart(targetCatch.Id) end)
                    if success then
                        if isStarted == false then task.wait(5) else
                            task.wait(1.5)
                            pcall(function() MonsterSystem:ClientCatchMonsterComplete(targetCatch.Id) end)
                        end
                    end
                end)
            end
        end
    end)

    -- ==========================================
    -- 3. LUỒNG CHẠY NỀN: AUTO FARM DỊCH CHUYỂN
    -- ==========================================
    task.spawn(function()
        local function GetAttackRemote()
            local s, r = pcall(function()
                return ReplicatedStorage:FindFirstChild("CommonLibrary")
                    and ReplicatedStorage.CommonLibrary:FindFirstChild("Tool")
                    and ReplicatedStorage.CommonLibrary.Tool:FindFirstChild("RemoteManager")
                    and ReplicatedStorage.CommonLibrary.Tool.RemoteManager:FindFirstChild("Funcs")
                    and ReplicatedStorage.CommonLibrary.Tool.RemoteManager.Funcs:FindFirstChild("DataPullFunc")
            end)
            return s and r or nil
        end

        while task.wait(0.1) do
            if not tpFarmSettings.enabled then continue end
            
            local controller = _G.SystemController
            if controller and (controller.IsPaused or controller.CurrentLock ~= nil) then task.wait(0.5) continue end
            if _G.PendingBossHunt or _G.PendingSpecialBoss or camSettings.isHealing then continue end

            local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
            local hum = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Humanoid")
            if not (hrp and hum and hum.Health > 0) then task.wait(1) continue end

            local centerPos = savedFarmPosition and savedFarmPosition.Position or hrp.Position
            
            -- Quét với mode "Attack"
            local validMonsters = Utils.SmartScanMonsters(centerPos, camSettings.attackRadius, "Attack")
            
            local closestMonster = nil
            local minDist = math.huge
            for _, mData in ipairs(validMonsters) do
                if mData.Distance < minDist then
                    minDist = mData.Distance
                    closestMonster = mData
                end
            end

            if closestMonster then
                local remote = GetAttackRemote()
                if remote then
                    local targetPos = closestMonster.Position + tpFarmSettings.distanceOffset
                    
                    -- Chống giật
                    if closestMonster.Distance > 3 then
                        hrp.CFrame = CFrame.new(targetPos)
                        hrp.CFrame = CFrame.lookAt(hrp.Position, closestMonster.Position)
                    end

                    pcall(function() remote:InvokeServer("MonsterAttackChannel", closestMonster.Id) end)
                    task.wait(tpFarmSettings.attackDelay + 0.1)
                end
            end
        end
    end)
        -- 3. LUỒNG HỒI MÁU PET
    local function CheckDeadPetsFromUI()
        local deadCount = 0
        local totalPets = 0
        pcall(function()
            local petListUI = LocalPlayer:FindFirstChild("PlayerGui") 
                and LocalPlayer.PlayerGui:FindFirstChild("MainGui") 
                and LocalPlayer.PlayerGui.MainGui:FindFirstChild("ScreenGui") 
                and LocalPlayer.PlayerGui.MainGui.ScreenGui:FindFirstChild("MainLeftView") 
                and LocalPlayer.PlayerGui.MainGui.ScreenGui.MainLeftView:FindFirstChild("PetList")
            if not petListUI then return end
            for _, cell in pairs(petListUI:GetChildren()) do
                if cell.Name == "ItemCell" then
                    local labName = cell:FindFirstChild("LabName", true)
                    if labName and labName.Text ~= "Label" and labName.Text ~= "" then
                        totalPets = totalPets + 1
                        local hp0 = cell:FindFirstChild("hp0", true)
                        if hp0 then
                            local bgColor = hp0.BackgroundColor3
                            if bgColor and (bgColor.R > 0.8 and bgColor.G < 0.3 and bgColor.B < 0.3) then
                                deadCount = deadCount + 1
                            end
                        else
                            deadCount = deadCount + 1
                        end
                    end
                end
            end
        end)
        return deadCount, totalPets
    end



    task.spawn(function()
        while task.wait(1) do
            local controller = _G.SystemController
            
            -- Kiểm tra Dừng toàn hệ thống
            if controller and controller.IsPaused then continue end

            -- ✅ THÊM ĐOẠN NÀY: Nếu hệ thống đang bận việc khác (Boss Hunt, Dungeon...), Pet tạm thời "nhịn" không đi hồi máu
            if controller and controller.CurrentLock ~= nil then 
                continue 
            end

            -- Tiếp tục logic cũ...
            if not camSettings.autoHeal or camSettings.isHealing then continue end
            if LocalPlayer:GetAttribute("RidePetId") ~= nil then continue end
            
            -- ... [Phần code phía dưới giữ nguyên] ...
            
            local deadCount, totalPets = CheckDeadPetsFromUI()
            if totalPets > 0 and deadCount >= camSettings.healThreshold then
                camSettings.isHealing = true
                local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
                if hrp then
                    camSettings.savedPosition = savedFarmPosition or hrp.CFrame
                    local recoverPos = FindRecoverPosition()
                    if recoverPos then
                        game.StarterGui:SetCore("SendNotification", { Title = "🔄 Auto Heal", Text = "Pet đã kiệt sức. Tiến hành Teleport hồi máu...", Duration = 3 })
                        local plat = Instance.new("Part")
                        plat.Size = Vector3.new(20, 1, 20); plat.Position = recoverPos - Vector3.new(0, 3, 0)
                        plat.Anchored = true; plat.Transparency = 1; plat.CanCollide = true; plat.Parent = Workspace
                        
                        SafeTeleport(CFrame.new(recoverPos), 5)
                        task.wait(10)
                        plat:Destroy()
                        
                        if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                            SafeTeleport(camSettings.savedPosition, 5)
                        end
                        game.StarterGui:SetCore("SendNotification", { Title = "✅ Hoàn Tất", Text = "Đã khôi phục sinh lực Pet và về bãi farm!", Duration = 3 })
                    end
                end
                camSettings.isHealing = false
                camSettings.savedPosition = nil
            end
        end
    end)

-- ==========================================
    -- LUỒNG CHẠY NỀN: AUTO FARM DỊCH CHUYỂN
    -- ==========================================
    task.spawn(function()
        local function GetAttackRemote()
            local success, remote = pcall(function()
                return ReplicatedStorage:FindFirstChild("CommonLibrary")
                    and ReplicatedStorage.CommonLibrary:FindFirstChild("Tool")
                    and ReplicatedStorage.CommonLibrary.Tool:FindFirstChild("RemoteManager")
                    and ReplicatedStorage.CommonLibrary.Tool.RemoteManager:FindFirstChild("Funcs")
                    and ReplicatedStorage.CommonLibrary.Tool.RemoteManager.Funcs:FindFirstChild("DataPullFunc")
            end)
            return success and remote or nil
        end

        while task.wait(0.1) do
            if not tpFarmSettings.enabled then continue end
            
            local controller = _G.SystemController
            
            -- ✅ CHẶN NGAY NẾU HỆ THỐNG ĐANG BẬN HOẶC BOSS CHUẨN BỊ RA
            if controller and (controller.IsPaused or controller.CurrentLock ~= nil) then
                task.wait(0.5)
                continue
            end
            if _G.PendingBossHunt or _G.PendingSpecialBoss then continue end

            local char = LocalPlayer.Character
            local hrp = char and char:FindFirstChild("HumanoidRootPart")
            local hum = char and char:FindFirstChild("Humanoid")
            
            if not (hrp and hum and hum.Health > 0) then 
                task.wait(1); continue 
            end

            -- ✅ XÁC ĐỊNH PHẠM VI: Tâm bãi farm
            local centerPos = savedFarmPosition and savedFarmPosition.Position or hrp.Position

            -- ✅ LỌC QUÁI: Xử lý bộ lọc cho Utils.ScanMonsters
            local scanKeywords = camSettings.targetMonsterNames
            if #scanKeywords == 1 and scanKeywords[1] == "Tất cả quái" then 
                scanKeywords = {} -- Gửi mảng rỗng để quét tất cả
            end

            local allMonsters = Utils.ScanMonsters(scanKeywords)
            if not allMonsters or #allMonsters == 0 then 
                task.wait(0.5)
                continue 
            end

            -- 4. Tìm con quái gần nhất để Teleport
            local closestMonster = nil
            local minDist = math.huge
            
            for _, m in ipairs(allMonsters) do
                local obj = m.Object
                local root = m.RootPart
                
                -- BƯỚC 1: Kiểm tra nhãn CatchEndTick (Có nhãn là đang bị bắt -> Bỏ qua)
                if obj:GetAttribute("CatchEndTick") ~= nil or (root and root:GetAttribute("CatchEndTick") ~= nil) then 
                    continue 
                end
                
                -- BƯỚC 2: MÁY ĐỌC MÁU (Chống lừa đảo từ Game)
                local isAlive = false
                local hum_monster = obj:FindFirstChildOfClass("Humanoid")
                
                if hum_monster then
                    -- Xử lý Quái thường
                    if hum_monster.Health >= 1 then isAlive = true end
                else
                    -- Xử lý Quái dùng StringValue
                    local hpVal = obj:FindFirstChild("Health")
                    if hpVal and hpVal:IsA("StringValue") then
                        local rawStr = hpVal.Value
                        local hpNum = 0
                        
                        -- Thử parse dạng thập phân thường (VD: "0,5" -> "0.5")
                        local tryDec = tonumber((string.gsub(rawStr, ",", ".")))
                        
                        if tryDec then
                            hpNum = tryDec
                        else
                            -- Giải mã định dạng BigNum của Game (VD: "9289000,-3")
                            local parts = string.split(rawStr, ",")
                            if #parts == 2 then
                                local base = tonumber(parts[1])
                                local exp = tonumber(parts[2])
                                if base and exp then
                                    hpNum = base * (10 ^ exp)
                                end
                            end
                        end
                        
                        -- Chỉ đánh khi Máu thật sự >= 1
                        if hpNum >= 1 then isAlive = true end
                    end
                end
                
                -- BƯỚC 3: Nếu sống & Trong phạm vi -> Đưa vào tầm ngắm
                if isAlive then
                    local distFromCenter = (centerPos - m.Position).Magnitude
                    if distFromCenter <= camSettings.attackRadius then
                        local distFromPlayer = (hrp.Position - m.Position).Magnitude
                        if distFromPlayer < minDist then
                            minDist = distFromPlayer
                            closestMonster = m
                        end
                    end
                end
            end
            
            -- 5. Thực thi: Bay tới và vả
            if closestMonster then
                local remote = GetAttackRemote()
                if remote then
                    -- Vị trí lý tưởng trên đầu quái
                    local targetPos = closestMonster.Position + tpFarmSettings.distanceOffset
                    local currentDist = (hrp.Position - targetPos).Magnitude

                    -- BỘ LỌC CHỐNG GIẬT
                    if currentDist > 3 then
                        hrp.CFrame = CFrame.new(targetPos)
                        hrp.CFrame = CFrame.lookAt(hrp.Position, closestMonster.Position)
                    end

                    -- Gửi lệnh tấn công
                    pcall(function() 
                        remote:InvokeServer("MonsterAttackChannel", closestMonster.Id) 
                    end)
                    
                    -- Tăng Delay lên một chút để chờ server cập nhật máu
                    task.wait(tpFarmSettings.attackDelay + 0.1)
                end
            end
            
        end
    end)

    -- ==========================================
    -- LUỒNG CHẠY NỀN: TỰ ĐỘNG BÁM BÃI FARM (SMART RETURN)
    -- ==========================================
    task.spawn(function()
        while task.wait(1) do
            local controller = _G.SystemController
            
            -- ✅ 1. CHẶN TOÀN TẬP: Nếu hệ thống bận, hoặc CÓ TÍN HIỆU BOSS CHUẨN BỊ XUẤT HIỆN
            if controller and (controller.IsPaused or controller.CurrentLock ~= nil) then continue end
            if _G.PendingBossHunt or _G.PendingSpecialBoss then continue end
            
            -- 2. Nếu đang đi hồi máu Pet -> Nhường luồng hồi máu làm xong việc
            if camSettings.isHealing then continue end

            -- 3. Chỉ kích hoạt khi đã lưu tọa độ và bật công tắc "Tự động về farm"
            if not camSettings.autoReturnToFarm or not savedFarmPosition then continue end

            -- 4. Phải có ít nhất 1 chế độ Farm đang chạy thì mới cần bám bãi
            if not (camSettings.autoAttack or camSettings.autoCatch or tpFarmSettings.enabled) then continue end

            local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
            if not hrp then continue end

            -- 5. Đo khoảng cách từ chỗ đang đứng đến tâm bãi Farm
            local dist = (hrp.Position - savedFarmPosition.Position).Magnitude
            
            -- Tính khoảng cách trôi dạt cho phép (Vượt ra khỏi bán kính farm 20 studs là xác định đi lạc)
            local maxAllowedDistance = camSettings.attackRadius + 20
            
            -- 6. Đi lạc -> Bế về!
            if dist > maxAllowedDistance then
                -- Check lại lần cuối ngay trước khi Teleport để đề phòng Boss vừa mới vạch đích ra
                if _G.PendingBossHunt or _G.PendingSpecialBoss or (controller and controller.CurrentLock ~= nil) then
                    continue
                end

                pcall(function()
                    game.StarterGui:SetCore("SendNotification", { Title = "📍 Smart Return", Text = "Hệ thống rảnh rỗi. Đang quay về bãi Farm...", Duration = 3 })
                end)
                
                SafeTeleport(savedFarmPosition, 5)
                task.wait(2)
            end
        end
    end)

end
