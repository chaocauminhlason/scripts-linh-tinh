-- ====================================================================
-- MODULE: AUTO RIFT SYSTEM V6 (PHÂN LẬP DYNAMIC/STATIC & CUSTOM TIME)
-- ====================================================================
return function(Window, Utils)
    local Players = game:GetService("Players")
    local Workspace = game:GetService("Workspace")
    local ReplicatedStorage = game:GetService("ReplicatedStorage")
    
    local LocalPlayer = Players.LocalPlayer
    local playerGui = LocalPlayer:WaitForChild("PlayerGui")

    -- ==========================================
    -- BIẾN CẤU HÌNH (STATE VARIABLES)
    -- ==========================================
    local autoRiftMaster = false 
    local scanDynamic = true    -- Công tắc quét Dynamic
    local scanStatic = true     -- Công tắc quét Static
    local riftRespawnTime = 300 -- Thời gian hồi sinh Static (Mặc định 300s = 5 Phút)

    local targetRiftId = nil 
    local targetRiftKey = nil 
    local currentTeamId = 0
    local riftsClearedCount = 0

    local dungeonStuckTimer = 0
    local dungeonLastPosition = nil
    local clearDelayTimer = 0 

    local completedRifts = {} 
    local catchingMemory = {} 

    local riftState = "SCANNING" 

    local DataPullFunc = nil

    -- ==========================================
    -- GIAO DIỆN UI (RAYFIELD)
    -- ==========================================
    local RiftTab = Window:CreateTab("Auto Rift", 4483362458)
    local RiftStatus = RiftTab:CreateLabel("STATUS: AUTO RIFT ĐANG TẮT")
    local ClearedLabel = RiftTab:CreateLabel("Rift đã dọn dẹp: 0")

    RiftTab:CreateSection("--- CÔNG TẮC ĐIỀU KHIỂN ---")
    
    RiftTab:CreateToggle({
        Name = "Công Tắc Tổng (Bật/Tắt Auto Rift)", 
        CurrentValue = false, 
        Flag = "AutoRiftMasterToggle",
        Callback = function(Value)
            autoRiftMaster = Value
            if not Value then 
                RiftStatus:Set("STATUS: TẮT")
                riftState = "SCANNING"
                targetRiftId = nil 
                targetRiftKey = nil
            else 
                RiftStatus:Set("STATUS: ĐANG CHUẨN BỊ QUÉT...") 
                riftState = "SCANNING" 
                clearDelayTimer = 0 
            end
        end
    })

    RiftTab:CreateToggle({
        Name = "🔍 Săn Cổng Tự Tạo (Dynamic Rift)", 
        CurrentValue = true, 
        Flag = "ScanDynamicToggle",
        Callback = function(Value) scanDynamic = Value end
    })

    RiftTab:CreateToggle({
        Name = "🔍 Săn Cổng Mặc Định (Static Rift)", 
        CurrentValue = true, 
        Flag = "ScanStaticToggle",
        Callback = function(Value) scanStatic = Value end
    })

    RiftTab:CreateSlider({
        Name = "Thời gian quét lại Static (Phút)",
        Range = {1, 30},
        Increment = 1,
        Suffix = "Phút",
        CurrentValue = 5,
        Flag = "StaticRespawnSlider",
        Callback = function(Value) 
            riftRespawnTime = Value * 60 -- Đổi từ phút sang giây
            print("⏳ [AUTO RIFT] Đã chỉnh thời gian vòng lặp Static thành: " .. riftRespawnTime .. " giây.")
        end,
    })

    RiftTab:CreateSection("--- CÔNG CỤ SỬA LỖI ---")

    RiftTab:CreateButton({ 
        Name = "Ép vào Combat Mode (Dùng nếu lỡ bật trong Rift)", 
        Callback = function() 
            riftState = "COMBAT"
            RiftStatus:Set("STATUS: Đã ép chuyển sang chế độ Chiến Đấu!")
        end 
    })

    RiftTab:CreateButton({ 
        Name = "Xóa Trí Nhớ Blacklist (Quét lại từ đầu)", 
        Callback = function() 
            completedRifts = {} 
            riftsClearedCount = 0 
            ClearedLabel:Set("Rift đã dọn dẹp: 0") 
            print("🔄 [AUTO RIFT] Đã reset toàn bộ trí nhớ Blacklist.")
        end 
    })

    -- ==========================================
    -- VÒNG LẶP XỬ LÝ CHÍNH
    -- ==========================================
    task.spawn(function()
        DataPullFunc = ReplicatedStorage:WaitForChild("CommonLibrary"):WaitForChild("Tool"):WaitForChild("RemoteManager"):WaitForChild("Funcs"):WaitForChild("DataPullFunc")

        while task.wait(0.5) do
            if not autoRiftMaster then continue end

            local char = LocalPlayer.Character
            local hrp = char and char:FindFirstChild("HumanoidRootPart")
            local humanoid = char and char:FindFirstChild("Humanoid")
            if not (hrp and humanoid and humanoid.Health > 0) then continue end

            -- [BỘ LỌC CHỐNG XUNG ĐỘT ABYSS TRIAL]
            local labStage = Utils.FindUIElementByName(playerGui, "LabStage")
            if labStage and labStage.Text ~= "Unknown" and labStage.Text ~= "" then
                RiftStatus:Set("STATUS: Đang trong Abyss Trial - Tạm nghỉ!")
                continue 
            end

            -- QUÉT QUÁI VẬT (CHỈ QUÉT KHI Ở TRONG TRẬN)
            local monstersFolder = Workspace:FindFirstChild("Monsters")
            local hasAliveMonsters, deadMonsters, bestTarget, closestDist2D = false, {}, nil, math.huge
            local unhandledDeadCount = 0
            
            if riftState == "COMBAT" or riftState == "EXITING" then
                hasAliveMonsters, deadMonsters, bestTarget, closestDist2D = Utils.ScanMonsters(monstersFolder, hrp)
                for _, m in pairs(deadMonsters) do
                    local id = tonumber(m.Name:match("Monster_(%d+)"))
                    if id and not catchingMemory[id] then
                        unhandledDeadCount = unhandledDeadCount + 1
                    end
                end
            end

            -- ==========================================
            -- GIAI ĐOẠN 1: SCANNING (TÌM CỔNG Ở SẢNH)
            -- ==========================================
            if riftState == "SCANNING" then
                -- Cập nhật trạng thái hiển thị
                local scanningText = "Radar đang dò: "
                if scanDynamic and scanStatic then scanningText = scanningText .. "Dynamic + Static..."
                elseif scanDynamic then scanningText = scanningText .. "Chỉ Dynamic..."
                elseif scanStatic then scanningText = scanningText .. "Chỉ Static..."
                else scanningText = "Chưa chọn loại cổng nào!" end
                
                RiftStatus:Set("STATUS: " .. scanningText)

                -- Bỏ qua nếu không bật công tắc săn nào
                if not scanDynamic and not scanStatic then
                    task.wait(1)
                    continue
                end

                local foundNewRift = false
                local currentTime = os.time()
                
                -- GARBAGE COLLECTOR
                for key, timeCleared in pairs(completedRifts) do
                    local isDynamic = type(key) == "string" and string.find(key, "Dynamic")
                    local expireTime = isDynamic and 120 or riftRespawnTime
                    if currentTime - timeCleared > expireTime then
                        completedRifts[key] = nil 
                    end
                end
                
                -- 1A. TÌM CỔNG DYNAMIC
                if scanDynamic then
                    local dynamicFolder = Workspace:FindFirstChild("DynamicDungeon")
                    if dynamicFolder then
                        for _, obj in pairs(dynamicFolder:GetChildren()) do
                            local idStr = string.match(obj.Name, "Dungeon_(%d+)")
                            if idStr then
                                local id = tonumber(idStr)
                                
                                local portalName = "UnknownPortal"
                                for _, child in pairs(obj:GetChildren()) do
                                    if string.match(child.Name, "Portal") then portalName = child.Name break end
                                end
                                local dynamicKey = "Dynamic_" .. idStr .. "_" .. portalName
                                
                                if not completedRifts[dynamicKey] then
                                    targetRiftId = id
                                    targetRiftKey = dynamicKey 
                                    foundNewRift = true
                                    RiftStatus:Set("STATUS: Khóa mục tiêu Dynamic: " .. dynamicKey)
                                    
                                    local doorPos = nil
                                    if obj:IsA("Model") and obj.PrimaryPart then doorPos = obj.PrimaryPart.Position
                                    elseif obj:IsA("BasePart") then doorPos = obj.Position
                                    else
                                        for _, desc in pairs(obj:GetDescendants()) do
                                            if desc:IsA("BasePart") then doorPos = desc.Position break end
                                        end
                                    end
                                    if not doorPos and obj:IsA("Model") then doorPos = obj:GetPivot().Position end
                                    
                                    if doorPos and (hrp.Position - doorPos).Magnitude > 20 then
                                        hrp.CFrame = CFrame.new(doorPos + Vector3.new(0, 3, 5))
                                        task.wait(1.5)
                                    end

                                    currentTeamId = os.time()
                                    pcall(function() DataPullFunc:InvokeServer("DungeonCreateTeamChannel", id, currentTeamId) end)
                                    task.wait(1.5)
                                    pcall(function() DataPullFunc:InvokeServer("DungeonStartChannel", id, currentTeamId) end)
                                    task.wait(3.5)
                                    
                                    dungeonStuckTimer = 0
                                    riftState = "COMBAT"
                                    break
                                end
                            end
                        end
                    end
                end

                -- 1B. TÌM CỔNG STATIC
                if not foundNewRift and scanStatic then
                    local areaFolder = Workspace:FindFirstChild("Area")
                    if areaFolder then
                        for _, mapFolder in pairs(areaFolder:GetChildren()) do
                            local serverZone = mapFolder:FindFirstChild("ServerZone")
                            if serverZone and serverZone:FindFirstChild("Dungeon") then
                                for _, obj in pairs(serverZone.Dungeon:GetChildren()) do
                                    local idStr = string.match(obj.Name, "Dungeon_(%d+)")
                                    if idStr then
                                        local id = tonumber(idStr)
                                        
                                        -- [BỘ LỌC CHỐNG LỪA]: BỎ QUA CỔNG BỊ ẨN / KHÓA
                                        local isHidden = false
                                        local guiNode = obj:FindFirstChild("GUI")
                                        
                                        if guiNode then
                                            pcall(function()
                                                if guiNode:IsA("BillboardGui") or guiNode:IsA("SurfaceGui") then
                                                    if not guiNode.Enabled then isHidden = true end
                                                else
                                                    if guiNode.Visible == false then isHidden = true end
                                                end
                                            end)
                                        end
                                        if isHidden then continue end
                                        
                                        -- TRA SỔ ĐEN (BLACKLIST) DỰA THEO THỜI GIAN TRÊN SLIDER
                                        if not completedRifts[id] or (currentTime - completedRifts[id] > riftRespawnTime) then
                                            targetRiftId = id
                                            targetRiftKey = id
                                            foundNewRift = true
                                            RiftStatus:Set("STATUS: Khóa mục tiêu Static ID: " .. id)
                                            
                                            local doorPos = obj:IsA("Model") and obj:GetPivot().Position or obj.Position
                                            
                                            if doorPos and (hrp.Position - doorPos).Magnitude > 20 then
                                                hrp.CFrame = CFrame.new(doorPos + Vector3.new(0, 3, 5))
                                                task.wait(1.5)
                                            end

                                            currentTeamId = os.time()
                                            pcall(function() DataPullFunc:InvokeServer("DungeonCreateTeamChannel", id, currentTeamId) end)
                                            task.wait(1.5)
                                            pcall(function() DataPullFunc:InvokeServer("DungeonStartChannel", id, currentTeamId) end)
                                            task.wait(3.5)
                                            
                                            dungeonStuckTimer = 0
                                            riftState = "COMBAT"
                                            break
                                        end
                                    end
                                end
                                if foundNewRift then break end
                            end
                        end
                    end
                end

                if not foundNewRift then
                    RiftStatus:Set("STATUS: Đã quét xong. Đang đợi cổng xuất hiện...")
                    task.wait(2)
                end

            -- ==========================================
            -- GIAI ĐOẠN 2: COMBAT (TRONG TRẬN)
            -- ==========================================
            elseif riftState == "COMBAT" then
                clearDelayTimer = 0 

                if unhandledDeadCount > 0 then
                    RiftStatus:Set("STATUS: Đang ném bóng bắt quái...")
                    for _, m in pairs(deadMonsters) do
                        local id = tonumber(m.Name:match("Monster_(%d+)"))
                        if id and not catchingMemory[id] then
                            catchingMemory[id] = true
                            Utils.CatchMonster(DataPullFunc, id) 
                        end
                    end
                elseif hasAliveMonsters then
                    RiftStatus:Set("STATUS: Đang trong trận - Chiến đấu!")
                    if bestTarget then
                        local targetPos = bestTarget:IsA("Model") and bestTarget.PrimaryPart and bestTarget.PrimaryPart.Position or bestTarget.Position
                        
                        if closestDist2D > 40 then
                            Utils.ToggleMount(DataPullFunc, true)
                            task.wait(0.2)
                            hrp.CFrame = CFrame.new(targetPos + Vector3.new(0, 3, 4))
                            task.wait(0.2)
                            humanoid:MoveTo(hrp.Position)
                            dungeonLastPosition = hrp.Position
                            dungeonStuckTimer = 0
                            Utils.SmartDismount(LocalPlayer, DataPullFunc)
                        elseif closestDist2D > 12 then
                            humanoid:MoveTo(targetPos)
                            if dungeonLastPosition and (hrp.Position - dungeonLastPosition).Magnitude < 1 then
                                dungeonStuckTimer = dungeonStuckTimer + 0.5
                                if dungeonStuckTimer >= 3 then hrp.CFrame = CFrame.new(targetPos + Vector3.new(0, 3, 3)); dungeonStuckTimer = 0 end
                            else dungeonStuckTimer = 0 end
                            dungeonLastPosition = hrp.Position
                        else
                            humanoid:MoveTo(hrp.Position) 
                            dungeonStuckTimer = 0 
                            Utils.SmartDismount(LocalPlayer, DataPullFunc)
                            
                            local id = tonumber(bestTarget.Name:match("Monster_(%d+)"))
                            if id then Utils.AttackMonster(DataPullFunc, id) end 
                        end
                    end
                else
                    riftState = "EXITING"
                end

            -- ==========================================
            -- GIAI ĐOẠN 3: EXITING (THOÁT TRẬN VÀ QUAY VỀ SẢNH)
            -- ==========================================
            elseif riftState == "EXITING" then
                clearDelayTimer = clearDelayTimer + 0.5
                RiftStatus:Set("STATUS: Đợi Server thu thập Pet... (" .. clearDelayTimer .. "s)")

                if #deadMonsters == 0 or clearDelayTimer >= 4 then
                    RiftStatus:Set("STATUS: Hoàn thành! Đang thoát...")
                    Utils.LeaveMatch(DataPullFunc, targetRiftId, currentTeamId)

                    if targetRiftKey then
                        completedRifts[targetRiftKey] = os.time()
                        riftsClearedCount = riftsClearedCount + 1
                        ClearedLabel:Set("Rift đã dọn dẹp: " .. riftsClearedCount)
                        print(string.format("🚪 [AUTO RIFT] Đã thoát trận! Ghi nhớ [%s] vào Blacklist.", tostring(targetRiftKey)))
                    end

                    targetRiftId = nil 
                    targetRiftKey = nil
                    catchingMemory = {} 
                    clearDelayTimer = 0
                    
                    riftState = "SCANNING" 
                    task.wait(3)
                end
            end
        end
    end)
end
