-- ====================================================================
-- MODULE: AUTO RIFT SYSTEM V8.1 (END-TICK SYNC & INSTANT EXIT)
-- ====================================================================
return function(Window, Utils)
    local Players = game:GetService("Players")
    local Workspace = game:GetService("Workspace")
    local ReplicatedStorage = game:GetService("ReplicatedStorage")
    
    local LocalPlayer = Players.LocalPlayer
    local playerGui = LocalPlayer:WaitForChild("PlayerGui")

    local autoRiftMaster = false 
    local scanDynamic = true    
    local scanStatic = true     
    
    local targetRiftId = nil 
    local targetRiftKey = nil 
    local targetEndTick = 0 -- Thay thế biến đếm tay bằng EndTick của Server
    local currentTeamId = 0
    local riftsClearedCount = 0
    
    local dungeonStuckTimer = 0
    local dungeonLastPosition = nil

    local completedRifts = {} 
    local catchingMemory = {} 
    local riftState = "SCANNING" 

    local DataPullFunc = nil

    local RiftTab = Window:CreateTab("Auto Rift", 4483362458)
    local RiftStatus = RiftTab:CreateLabel("STATUS: AUTO RIFT ĐANG TẮT")
    local ClearedLabel = RiftTab:CreateLabel("Rift đã dọn dẹp: 0")

    RiftTab:CreateSection("--- CÔNG TẮC ĐIỀU KHIỂN ---")
    RiftTab:CreateToggle({
        Name = "Công Tắc Tổng (Bật/Tắt Auto Rift)", CurrentValue = false, Flag = "AutoRiftMasterToggle",
        Callback = function(Value)
            autoRiftMaster = Value
            if not Value then 
                RiftStatus:Set("STATUS: TẮT"); riftState = "SCANNING"; targetRiftId = nil; targetRiftKey = nil
            else 
                RiftStatus:Set("STATUS: ĐANG CHUẨN BỊ QUÉT..."); riftState = "SCANNING"
            end
        end
    })

    RiftTab:CreateToggle({ Name = "🔍 Săn Cổng Tự Tạo (Dynamic)", CurrentValue = true, Flag = "ScanDynamicToggle", Callback = function(Value) scanDynamic = Value end })
    RiftTab:CreateToggle({ Name = "🔍 Săn Cổng Mặc Định (Static)", CurrentValue = true, Flag = "ScanStaticToggle", Callback = function(Value) scanStatic = Value end })

    RiftTab:CreateSection("--- CÔNG CỤ SỬA LỖI ---")
    RiftTab:CreateButton({ Name = "Ép vào Combat Mode (Dùng lỡ bật trong Rift)", Callback = function() riftState = "COMBAT" RiftStatus:Set("STATUS: Đã ép Chiến Đấu!") end })
    RiftTab:CreateButton({ Name = "Xóa Trí Nhớ Blacklist (Quét lại từ đầu)", Callback = function() completedRifts = {} riftsClearedCount = 0 ClearedLabel:Set("Rift đã dọn dẹp: 0") end })

    -- ==========================================
    -- HÀM LẤY TỌA ĐỘ 
    -- ==========================================
    local function GetPortalPosition(obj)
        local guiNode = obj:FindFirstChild("GUI")
        if guiNode then 
            if guiNode:IsA("Attachment") then return guiNode.WorldPosition end
            local pos
            local success = pcall(function() pos = guiNode:GetPivot().Position end)
            if success and pos then return pos end
        end
        if obj:IsA("Model") and obj.PrimaryPart then return obj.PrimaryPart.Position end
        return obj:GetPivot().Position 
    end

    task.spawn(function()
        DataPullFunc = ReplicatedStorage:WaitForChild("CommonLibrary"):WaitForChild("Tool"):WaitForChild("RemoteManager"):WaitForChild("Funcs"):WaitForChild("DataPullFunc")

        while task.wait(0.5) do
            if not autoRiftMaster then continue end

            local char = LocalPlayer.Character
            local hrp = char and char:FindFirstChild("HumanoidRootPart")
            local humanoid = char and char:FindFirstChild("Humanoid")
            if not (hrp and humanoid and humanoid.Health > 0) then continue end

            local labStage = Utils.FindUIElementByName(playerGui, "LabStage")
            if labStage and labStage.Text ~= "Unknown" and labStage.Text ~= "" then
                RiftStatus:Set("STATUS: Đang trong Abyss Trial - Tạm nghỉ!")
                continue 
            end

            local monstersFolder = Workspace:FindFirstChild("Monsters")
            local hasAliveMonsters, deadMonsters, bestTarget, closestDist2D = false, {}, nil, math.huge
            local unhandledDeadCount = 0
            
            if riftState == "COMBAT" or riftState == "EXITING" then
                hasAliveMonsters, deadMonsters, bestTarget, closestDist2D = Utils.ScanMonsters(monstersFolder, hrp)
                for _, m in pairs(deadMonsters) do
                    local id = tonumber(m.Name:match("Monster_(%d+)"))
                    if id and not catchingMemory[id] then unhandledDeadCount = unhandledDeadCount + 1 end
                end
            end

            if riftState == "SCANNING" then
                RiftStatus:Set("STATUS: Radar đang quét tìm Rift...")
                if not scanDynamic and not scanStatic then task.wait(1) continue end

                local currentTime = os.time()
                
                -- ĐỒNG BỘ RÁC THEO END TICK CỦA SERVER
                for key, expireTime in pairs(completedRifts) do
                    if currentTime > expireTime then completedRifts[key] = nil end
                end
                
                local bestPortal = nil
                local shortestDistance = math.huge

                -- ==========================================
                -- BƯỚC 1: QUÉT CỔNG DYNAMIC
                -- ==========================================
                if scanDynamic then
                    local dynamicFolder = Workspace:FindFirstChild("DynamicDungeon")
                    if dynamicFolder then
                        for _, obj in pairs(dynamicFolder:GetChildren()) do
                            local idStr = string.match(obj.Name, "Dungeon_(%d+)")
                            if idStr then
                                local id = tonumber(idStr)
                                local properTeamId = obj:GetAttribute("DungeonStartTick")
                                
                                if properTeamId and properTeamId > 0 then
                                    local dynamicKey = "Dyn_" .. idStr .. "_" .. properTeamId
                                    
                                    if not completedRifts[dynamicKey] then
                                        local doorPos = GetPortalPosition(obj)
                                        local dist = (hrp.Position - doorPos).Magnitude
                                        
                                        if dist < shortestDistance then
                                            shortestDistance = dist
                                            -- Lấy EndTick, dự phòng cộng thêm 5 phút nếu lỗi game
                                            local endTick = obj:GetAttribute("DungeonEndTick") or (os.time() + 300)
                                            bestPortal = {id = id, key = dynamicKey, pos = doorPos, teamId = properTeamId, eTick = endTick}
                                        end
                                    end
                                end
                            end
                        end
                    end
                end

                -- ==========================================
                -- BƯỚC 2: QUÉT CỔNG STATIC
                -- ==========================================
                if scanStatic and not bestPortal then
                    local areaFolder = Workspace:FindFirstChild("Area")
                    if areaFolder then
                        for _, mapFolder in pairs(areaFolder:GetChildren()) do
                            local arDungeon = mapFolder:FindFirstChild("Area") and mapFolder.Area:FindFirstChild("Dungeon")
                            if arDungeon then
                                for _, obj in pairs(arDungeon:GetChildren()) do
                                    local idStr = string.match(obj.Name, "Dungeon_(%d+)")
                                    if idStr then
                                        local id = tonumber(idStr)
                                        local properTeamId = obj:GetAttribute("DungeonStartTick")
                                        
                                        if properTeamId and properTeamId > 0 then
                                            local staticKey = "Stat_" .. idStr .. "_" .. properTeamId
                                            
                                            if not completedRifts[staticKey] then
                                                local doorPos = GetPortalPosition(obj)
                                                local dist = (hrp.Position - doorPos).Magnitude
                                                
                                                if dist < shortestDistance then
                                                    shortestDistance = dist
                                                    local endTick = obj:GetAttribute("DungeonEndTick") or (os.time() + 300)
                                                    bestPortal = {id = id, key = staticKey, pos = doorPos, teamId = properTeamId, eTick = endTick}
                                                end
                                            end
                                        end
                                    end
                                end
                            end
                        end
                    end
                end

                -- ==========================================
                -- BƯỚC 3: LAO VÀO CỔNG TỐT NHẤT
                -- ==========================================
                if bestPortal then
                    targetRiftId = bestPortal.id
                    targetRiftKey = bestPortal.key
                    targetEndTick = bestPortal.eTick
                    
                    RiftStatus:Set("STATUS: Đã khóa mục tiêu: " .. tostring(targetRiftId))
                    
                    hrp.CFrame = CFrame.new(bestPortal.pos + Vector3.new(0, 3, 0))
                    task.wait(0.2) 

                    currentTeamId = bestPortal.teamId
                    local createArgs = { "DungeonCreateTeamChannel", targetRiftId, currentTeamId }
                    local startArgs = { "DungeonStartChannel", targetRiftId, currentTeamId }

                    pcall(function()
                        DataPullFunc:InvokeServer(unpack(createArgs))
                        DataPullFunc:InvokeServer(unpack(startArgs))
                    end)

                    RiftStatus:Set("STATUS: Đang đợi Server dịch chuyển...")
                    task.wait(2.5) 
                    
                    local checkChar = LocalPlayer.Character
                    local checkHrp = checkChar and checkChar:FindFirstChild("HumanoidRootPart")
                    if checkHrp and (checkHrp.Position - bestPortal.pos).Magnitude < 50 then
                        RiftStatus:Set("STATUS: Bị Server chặn, bỏ qua cổng này!")
                        completedRifts[targetRiftKey] = os.time() + 15 
                        targetRiftId = nil; targetRiftKey = nil; riftState = "SCANNING"
                    else
                        dungeonStuckTimer = 0; riftState = "COMBAT"
                    end
                else
                    task.wait(0.5) 
                end

            elseif riftState == "COMBAT" then
                if unhandledDeadCount > 0 then
                    RiftStatus:Set("STATUS: Đang ném bóng bắt quái...")
                    for _, m in pairs(deadMonsters) do
                        local id = tonumber(m.Name:match("Monster_(%d+)"))
                        if id and not catchingMemory[id] then
                            catchingMemory[id] = true; Utils.CatchMonster(DataPullFunc, id) 
                        end
                    end
                elseif hasAliveMonsters then
                    RiftStatus:Set("STATUS: Đang trong trận - Chiến đấu!")
                    if bestTarget then
                        local targetPos = bestTarget:IsA("Model") and bestTarget.PrimaryPart and bestTarget.PrimaryPart.Position or bestTarget.Position
                        
                        if closestDist2D > 40 then
                            Utils.ToggleMount(DataPullFunc, true); task.wait(0.2)
                            hrp.CFrame = CFrame.new(targetPos + Vector3.new(0, 3, 4)); task.wait(0.2)
                            humanoid:MoveTo(hrp.Position); dungeonLastPosition = hrp.Position; dungeonStuckTimer = 0
                            Utils.SmartDismount(LocalPlayer, DataPullFunc)
                        elseif closestDist2D > 12 then
                            humanoid:MoveTo(targetPos)
                            if dungeonLastPosition and (hrp.Position - dungeonLastPosition).Magnitude < 1 then
                                dungeonStuckTimer = dungeonStuckTimer + 0.5
                                if dungeonStuckTimer >= 3 then hrp.CFrame = CFrame.new(targetPos + Vector3.new(0, 3, 3)); dungeonStuckTimer = 0 end
                            else dungeonStuckTimer = 0 end
                            dungeonLastPosition = hrp.Position
                        else
                            humanoid:MoveTo(hrp.Position); dungeonStuckTimer = 0; Utils.SmartDismount(LocalPlayer, DataPullFunc)
                            local id = tonumber(bestTarget.Name:match("Monster_(%d+)"))
                            if id then Utils.AttackMonster(DataPullFunc, id) end 
                        end
                    end
                else 
                    -- Hết quái sống, bắt xong toàn bộ quái chết -> Lập tức thoát
                    riftState = "EXITING" 
                end

            elseif riftState == "EXITING" then
                RiftStatus:Set("STATUS: Hoàn thành! Đang ép thoát siêu tốc...")
                
                -- KHÔNG CẦN CHỜ NỮA, BẮN REMOTE THOÁT NGAY LẬP TỨC
                pcall(function()
                    local leaveArenaArgs = { "ArenaLeaveChannel" }
                    DataPullFunc:InvokeServer(unpack(leaveArenaArgs))
                    
                    if targetRiftId and currentTeamId then
                        local leaveTeamArgs = { "DungeonLeaveTeamChannel", targetRiftId, currentTeamId }
                        DataPullFunc:InvokeServer(unpack(leaveTeamArgs))
                    end
                end)
                
                if targetRiftKey then
                    -- Ghi sổ đen bằng chính thời gian "Hết hạn" của cổng trên Server
                    completedRifts[targetRiftKey] = targetEndTick
                    riftsClearedCount = riftsClearedCount + 1
                    ClearedLabel:Set("Rift đã dọn dẹp: " .. riftsClearedCount)
                end
                
                targetRiftId = nil; targetRiftKey = nil; targetEndTick = 0
                catchingMemory = {}
                
                riftState = "SCANNING"; task.wait(0)
            end
        end
    end)
end
