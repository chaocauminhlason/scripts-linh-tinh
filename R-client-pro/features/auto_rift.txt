-- ====================================================================
-- MODULE: AUTO RIFT SYSTEM V4 (SỬA LỖI KẸT XÁC QUÁI & ÉP VÀO CỔNG)
-- ====================================================================
return function(Window, Utils)
    local Players = game:GetService("Players")
    local Workspace = game:GetService("Workspace")
    local ReplicatedStorage = game:GetService("ReplicatedStorage")
    
    local LocalPlayer = Players.LocalPlayer
    local playerGui = LocalPlayer:WaitForChild("PlayerGui")

    local autoRiftEnabled = false
    local targetRiftId = nil 
    local targetRiftKey = nil 
    local currentTeamId = 0
    local inRiftMatch = false
    local riftsClearedCount = 0

    local dungeonStuckTimer = 0
    local dungeonLastPosition = nil
    
    -- [BIẾN MỚI] Dùng để đếm ngược ép thoát trận
    local clearDelayTimer = 0 

    local completedRifts = {} 
    local riftRespawnTime = 300 
    local catchingMemory = {} 


    local RiftTab = Window:CreateTab("Auto Rift", 4483362458)
    local RiftStatus = RiftTab:CreateLabel("STATUS: AUTO RIFT ĐANG TẮT")
    local ClearedLabel = RiftTab:CreateLabel("Rift đã dọn dẹp: 0")

    RiftTab:CreateSection("--- HỆ THỐNG QUÉT BẢN ĐỒ ---")
    RiftTab:CreateToggle({
        Name = "Bật/Tắt Auto Quét & Đi Rift", CurrentValue = false, Flag = "AutoRiftMasterToggle",
        Callback = function(Value)
            autoRiftEnabled = Value
            if not Value then 
                RiftStatus:Set("STATUS: TẮT")
                inRiftMatch = false
                targetRiftId = nil 
            else 
                RiftStatus:Set("STATUS: ĐANG QUÉT MAP...") 
                clearDelayTimer = 0 -- Reset bộ đếm an toàn
            end
        end
    })
    RiftTab:CreateButton({ 
        Name = "Xóa Trí Nhớ Blacklist (Sửa lỗi kẹt cổng)", 
        Callback = function() 
            completedRifts = {} 
            riftsClearedCount = 0 
            ClearedLabel:Set("Rift đã dọn dẹp: 0") 
            print("🔄 [AUTO RIFT] Đã reset toàn bộ trí nhớ Blacklist.")
        end 
    })

    task.spawn(function()
        -- DI CHUYỂN DataPullFunc VÀO ĐÂY (Nằm trong luồng chạy ngầm)
        local DataPullFunc = ReplicatedStorage:WaitForChild("CommonLibrary"):WaitForChild("Tool"):WaitForChild("RemoteManager"):WaitForChild("Funcs"):WaitForChild("DataPullFunc")

        while task.wait(0.5) do
            if not autoRiftEnabled then continue end
            local char = LocalPlayer.Character
            local hrp = char and char:FindFirstChild("HumanoidRootPart")
            local humanoid = char and char:FindFirstChild("Humanoid")
            if not (hrp and humanoid and humanoid.Health > 0) then continue end

            -- [BỘ LỌC CHỐNG XUNG ĐỘT ABYSS TRIAL]
            local labStage = Utils.FindUIElementByName(playerGui, "LabStage")
            if labStage and labStage.Text ~= "Unknown" and labStage.Text ~= "" then
                RiftStatus:Set("STATUS: Đang trong Abyss Trial - Tạm nghỉ!")
                continue 
            end

            local monstersFolder = Workspace:FindFirstChild("Monsters")
            local hasAliveMonsters, deadMonsters, bestTarget, closestDist2D = Utils.ScanMonsters(monstersFolder, hrp)

            -- Đếm xem có con quái chết nào CHƯA ĐƯỢC NÉM BÓNG không?
            local unhandledDeadCount = 0
            for _, m in pairs(deadMonsters) do
                local id = tonumber(m.Name:match("Monster_(%d+)"))
                if id and not catchingMemory[id] then
                    unhandledDeadCount = unhandledDeadCount + 1
                end
            end

            -- ==========================================
            -- 1. BẮT QUÁI TRẮNG MÁU
            -- ==========================================
            if unhandledDeadCount > 0 then
                RiftStatus:Set("STATUS: Đang ném bóng bắt quái...")
                for _, m in pairs(deadMonsters) do
                    local id = tonumber(m.Name:match("Monster_(%d+)"))
                    if id and not catchingMemory[id] then
                        catchingMemory[id] = true
                        Utils.CatchMonster(DataPullFunc, id) 
                    end
                end
            end

            -- ==========================================
            -- 2. ĐÁNH QUÁI CÒN SỐNG
            -- ==========================================
            if hasAliveMonsters then
                inRiftMatch = true
                clearDelayTimer = 0 -- Reset bộ đếm thoát phòng vì vẫn đang đánh
                
                if unhandledDeadCount == 0 then RiftStatus:Set("STATUS: Đang trong trận - Chiến đấu!") end

                if bestTarget then
                    local targetPos = bestTarget:IsA("Model") and bestTarget.PrimaryPart and bestTarget.PrimaryPart.Position or bestTarget.Position
                    
                    if closestDist2D > 40 then
                        Utils.ToggleMount(DataPullFunc, true)
                        task.wait(0.2)
                        hrp.CFrame = CFrame.new(targetPos + Vector3.new(0, 3, 4))
                        task.wait(0.2)
                        humanoid:MoveTo(hrp.Position)
                        dungeonLastPosition = hrp.Position
                        dungeonStuckTimer = 0
                        Utils.SmartDismount(LocalPlayer, DataPullFunc)
                    elseif closestDist2D > 12 then
                        humanoid:MoveTo(targetPos)
                        if dungeonLastPosition and (hrp.Position - dungeonLastPosition).Magnitude < 1 then
                            dungeonStuckTimer = dungeonStuckTimer + 0.5
                            if dungeonStuckTimer >= 3 then hrp.CFrame = CFrame.new(targetPos + Vector3.new(0, 3, 3)); dungeonStuckTimer = 0 end
                        else dungeonStuckTimer = 0 end
                        dungeonLastPosition = hrp.Position
                    else
                        humanoid:MoveTo(hrp.Position) 
                        dungeonStuckTimer = 0 
                        Utils.SmartDismount(LocalPlayer, DataPullFunc)
                        
                        local id = tonumber(bestTarget.Name:match("Monster_(%d+)"))
                        if id then Utils.AttackMonster(DataPullFunc, id) end 
                    end
                end

            -- ==========================================
            -- 3. THOÁT TRẬN (KHI QUÁI CHẾT HẾT VÀ NÉM BÓNG XONG)
            -- ==========================================
            elseif inRiftMatch and not hasAliveMonsters and unhandledDeadCount == 0 then
                clearDelayTimer = clearDelayTimer + 0.5
                RiftStatus:Set("STATUS: Đợi Server thu thập Pet... (" .. clearDelayTimer .. "s)")

                -- [FIX LỖI KẸT]: Thoát nếu xác quái đã biến mất HOẶC đã đợi quá 4 giây!
                if #deadMonsters == 0 or clearDelayTimer >= 4 then
                    RiftStatus:Set("STATUS: Hoàn thành! Đang thoát...")
                    Utils.LeaveMatch(DataPullFunc, targetRiftId, currentTeamId)

                    if targetRiftId then
                        local memoryKey = targetRiftKey or targetRiftId
                        completedRifts[memoryKey] = os.time()
                        riftsClearedCount = riftsClearedCount + 1
                        ClearedLabel:Set("Rift đã dọn dẹp: " .. riftsClearedCount)
                        print(string.format("🚪 [AUTO RIFT] Đã thoát trận! Ghi nhớ [%s] vào Blacklist.", tostring(memoryKey)))
                    end

                    inRiftMatch = false
                    targetRiftId = nil 
                    targetRiftKey = nil
                    catchingMemory = {} 
                    clearDelayTimer = 0
                    task.wait(3)
                end

            -- ==========================================
            -- 4. SẢNH CHỜ QUÉT DYNAMIC RIFT
            -- ==========================================
            else
                inRiftMatch = false -- Failsafe: Ép tắt trạng thái trận đấu nếu kẹt
                RiftStatus:Set("STATUS: Sảnh - Radar đang dò tìm Rift...")
                local foundNewRift = false
                local currentTime = os.time()
                
                -- GARBAGE COLLECTOR
                for key, timeCleared in pairs(completedRifts) do
                    local isDynamic = type(key) == "string" and string.find(key, "Dynamic")
                    local expireTime = isDynamic and 120 or riftRespawnTime
                    if currentTime - timeCleared > expireTime then
                        completedRifts[key] = nil 
                    end
                end
                
                local dynamicFolder = Workspace:FindFirstChild("DynamicDungeon")
                if dynamicFolder then
                    for _, obj in pairs(dynamicFolder:GetChildren()) do
                        local idStr = string.match(obj.Name, "Dungeon_(%d+)")
                        if idStr then
                            local id = tonumber(idStr)
                            
                            local portalName = "UnknownPortal"
                            for _, child in pairs(obj:GetChildren()) do
                                if string.match(child.Name, "Portal") then portalName = child.Name break end
                            end
                            local dynamicKey = "Dynamic_" .. idStr .. "_" .. portalName
                            
                            -- KIỂM TRA SỔ ĐEN VÀ IN RA LOG NẾU BỊ CHẶN
                            if completedRifts[dynamicKey] then
                                -- print("⚠️ Bỏ qua Rift: " .. dynamicKey .. " vì đang trong thời gian Cooldown!")
                            else
                                targetRiftId = id
                                targetRiftKey = dynamicKey 
                                foundNewRift = true
                                RiftStatus:Set("STATUS: Đã khóa mục tiêu: " .. dynamicKey)
                                
                                local doorPos = nil
                                if obj:IsA("Model") and obj.PrimaryPart then doorPos = obj.PrimaryPart.Position
                                elseif obj:IsA("BasePart") then doorPos = obj.Position
                                else
                                    for _, desc in pairs(obj:GetDescendants()) do
                                        if desc:IsA("BasePart") then doorPos = desc.Position break end
                                    end
                                end
                                if not doorPos and obj:IsA("Model") then doorPos = obj:GetPivot().Position end
                                
                                if doorPos and (hrp.Position - doorPos).Magnitude > 20 then
                                    hrp.CFrame = CFrame.new(doorPos + Vector3.new(0, 3, 5))
                                    task.wait(1.5)
                                end

                                currentTeamId = os.time()
                                pcall(function() DataPullFunc:InvokeServer("DungeonCreateTeamChannel", id, currentTeamId) end)
                                task.wait(1.5)
                                pcall(function() DataPullFunc:InvokeServer("DungeonStartChannel", id, currentTeamId) end)
                                task.wait(3.5)
                                break
                            end
                        end
                    end
                end

                if not foundNewRift then
                    RiftStatus:Set("STATUS: Đã dọn sạch Map. Đợi Rift mới xuất hiện...")
                    task.wait(2)
                end
            end
        end
    end)
end
