-- ====================================================================
-- MAIN SCRIPT: Khởi chạy R-Client Pro (BẢN CHỐNG CACHE GITHUB)
-- ====================================================================
local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

local Window = Rayfield:CreateWindow({
    Name = "R-Client Pro Tool v2.5",
    LoadingTitle = "Đang tải hệ thống Core...",
    LoadingSubtitle = "Dev by Sơn",
    ConfigurationSaving = { Enabled = true, FileName = "AutoFarmConfig" },
    KeySystem = false,
    DisableRayfieldPrompts = true,
    DisableBuildWarnings = true 
})

-- Tạo đuôi thời gian thực để XUYÊN THỦNG bộ nhớ đệm 5 phút của GitHub
local noCache = "?t=" .. tostring(os.time())

-- Hàm tải an toàn siêu việt
local function SafeLoad(url)
    local success, result = pcall(function() return game:HttpGet(url .. noCache) end)
    if success then
        local func, err = loadstring(result)
        if func then 
            return func() 
        else 
            warn("❌ Lỗi Cú Pháp tại: " .. url .. "\n" .. tostring(err)) 
        end
    else
        warn("❌ Không kết nối được: " .. url)
    end
    return nil
end

-- 1. Kéo các file Core
local Utils = SafeLoad('https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/R-client-pro/core/utilities.txt')
local WebhookModule = SafeLoad('https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/R-client-pro/core/webhook.txt')


-- 2. Kéo các Features
if Utils then
    local t = SafeLoad('https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/R-client-pro/features/tracker.txt')
    if t then t(Window, Utils) end

    local sg = SafeLoad('https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/R-client-pro/features/shops_and_guis.txt')
    if sg then sg(Window, Utils) end

    local r = SafeLoad('https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/R-client-pro/features/farm.txt')
    if r then r(Window, Utils) end
    
    -- Tải Module Săn Boss
    local b = SafeLoad('https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/R-client-pro/features/boss_hunt.txt')
    if b then b(Window, Utils) end
    
    -- Tải Module Quản Lý Server
    local sm = SafeLoad('https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/R-client-pro/features/server_manager.txt')
    if sm then sm(Window, Utils) end
    
    -- Load Hầm Ngục ĐỘC LẬP (Không bị phụ thuộc vào Webhook nữa)
    local d = SafeLoad('https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/R-client-pro/features/auto_dungeon.txt')
    if d then d(Window, Utils, WebhookModule) end
    
    -- [ĐÃ SỬA LỖI TÊN FILE]: auto_hatch.txt -> hatch_egg.txt
    local h = SafeLoad('https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/R-client-pro/features/hatch_egg.txt')
    if h then h(Window, Utils) end

    local rift = SafeLoad('https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/R-client-pro/features/auto_rift.txt')
    if rift then rift(Window, Utils) end
    
    -- [ĐÃ SỬA LỖI MẤT BIẾN O]: Thêm "local o ="
    local o = SafeLoad('https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/R-client-pro/features/optimization.txt')
    if o then o(Window, Utils) end
end

if WebhookModule and WebhookModule.InitTab then
    WebhookModule.InitTab(Window, Rayfield)
end

Rayfield:LoadConfiguration()
Rayfield:Notify({ Title = "Thành Công!", Content = "Hệ thống R-Client Pro đã nạp đầy đủ Modules!", Duration = 5 })
