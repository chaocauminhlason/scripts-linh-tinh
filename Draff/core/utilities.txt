local VirtualInputManager = game:GetService("VirtualInputManager")
local UserInputService = game:GetService("UserInputService")

local Utils = {}

-- [1] HÀM NẠP FILE AN TOÀN (SAFELOAD)
function Utils.SafeLoad(filePath)
    local content = nil
    
    -- 1. Thử đọc file cục bộ (đường dẫn tuyệt đối hoặc tương đối)
    local possiblePaths = {
        "d:/scripts/scripts-linh-tinh/Draff/" .. filePath,
        "Draff/" .. filePath
    }
    
    for _, p in ipairs(possiblePaths) do
        if isfile and isfile(p) then
            local ok, data = pcall(readfile, p)
            if ok and data then
                content = data
                break
            end
        end
    end
    
    -- 2. Thử đọc từ Localhost Web Server (nếu đang chạy Local Dev)
    if not content then
        local localUrl = "http://127.0.0.1:8002/Draff/" .. filePath .. "?t=" .. tostring(os.time())
        local ok, data = pcall(game.HttpGet, game, localUrl)
        if ok and data and not string.find(data, "404") then content = data end
    end

    -- 3. Nếu không thấy file cục bộ / localhost, nạp qua GitHub repository (online)
    if not content then
        local rawUrl = "https://raw.githubusercontent.com/chaocauminhlason/scripts-linh-tinh/main/Draff/" .. filePath
        local ok, data = pcall(game.HttpGet, game, rawUrl)
        if ok and data then content = data end
    end
    
    if content then
        local func, err = loadstring(content)
        if func then
            local success, result = pcall(func)
            if success then
                return result
            else
                warn("⚠️ [SafeLoad] Lỗi thực thi module tại: " .. filePath .. "\n" .. tostring(result))
            end
        else
            warn("⚠️ [SafeLoad] Lỗi Cú Pháp tại: " .. filePath .. "\n" .. tostring(err))
        end
    else
        warn("⚠️ [SafeLoad] Không nạp được file: " .. filePath)
    end
    return nil
end

-- [2] LOGIC CLICK CHUỘT GIỚI HẠN TỐC ĐỘ (THROTTLED CLICK)
local lastClickTime = 0

function Utils.ThrottledClick(x, y, cooldown)
    cooldown = cooldown or 0.12
    local now = tick()
    if now - lastClickTime >= cooldown then
        lastClickTime = now
        VirtualInputManager:SendMouseButtonEvent(x, y, 0, true, game, 0)
        task.wait(0.01)
        VirtualInputManager:SendMouseButtonEvent(x, y, 0, false, game, 0)
    end
end

-- [3] HÀM KÉO THẢ GIAO DIỆN (MAKE DRAGGABLE)
function Utils.MakeDraggable(topbar, obj)
    local dragging, dragInput, dragStart, startPos = false, nil, nil, nil
    obj:SetAttribute("Locked", false)
    
    local function Update(input)
        if obj:GetAttribute("Locked") then return end
        local delta = input.Position - dragStart
        obj.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
    
    topbar.InputBegan:Connect(function(input)
        if input.UserInputType ~= Enum.UserInputType.MouseButton1 and input.UserInputType ~= Enum.UserInputType.Touch then return end
        dragging = not obj:GetAttribute("Locked")
        dragStart = input.Position
        startPos = obj.Position
        
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end)
    
    topbar.InputChanged:Connect(function(input)
        if not dragStart then return end
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)
    
    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then Update(input) end
    end)
end

return Utils
