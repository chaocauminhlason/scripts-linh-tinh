-- 1. Thử đọc file main.txt trực tiếp từ thư mục workspace của Executor (readfile)
if isfile and isfile("Draff/main.txt") then
    print("[ApolloHub | Local Dev] 📂 Phát hiện file Draff/main.txt cục bộ. Đang nạp qua readfile()...")
    local ok, content = pcall(readfile, "Draff/main.txt")
    if ok and content then
        local func, err = loadstring(content)
        if func then
            func()
            return
        else
            warn("❌ [Local Dev] Lỗi cú pháp trong Draff/main.txt:\n" .. tostring(err))
        end
    end
end

-- 2. Fallback: Tải qua Localhost Web Server (chạy `python -m http.server 8000` tại D:\scripts\scripts-linh-tinh)
local localUrl = "http://127.0.0.1:8000/Draff/main.txt?t=" .. tostring(os.time())
print("[ApolloHub | Local Dev] 🌐 Thử nạp từ Localhost: " .. localUrl)

local success, result = pcall(function() return game:HttpGet(localUrl) end)
if success and not string.find(result, "404") then
    local func, err = loadstring(result)
    if func then
        func()
        print("🚀 [ApolloHub | Local Dev] Nạp thành công từ Localhost Server!")
    else
        warn("❌ [Local Dev] Lỗi cú pháp từ Localhost Server:\n" .. tostring(err))
    end
else
    warn("❌ [Local Dev] Không kết nối được Localhost Server hoặc không thấy file Draff/main.txt!")
    warn("👉 Hướng dẫn: Mở PowerShell tại D:\\scripts\\scripts-linh-tinh và chạy lệnh: python -m http.server 8000")
end
