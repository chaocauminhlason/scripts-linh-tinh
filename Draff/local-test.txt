-- ==========================================
-- BƯỚC 2: HÀM TẢI LOCAL AN TOÀN (SAFE LOAD) - Y HỆT R-CLIENT-PRO
-- ==========================================
local function SafeLoad(filePath)
    local fullUrl = repoUrl .. filePath .. "?t=" .. tostring(os.time())
    
    -- Đọc file trực tiếp qua HTTP Localhost, tốc độ gần như bằng 0ms
    local success, result = pcall(function() return game:HttpGet(fullUrl) end)
    
    if success and not string.find(result, "404") then
        local func, err = loadstring(result)
        if func then 
            return func() 
        else 
            warn("❌ Lỗi Cú Pháp tại file local: " .. filePath .. "\n" .. tostring(err)) 
        end
    else
        warn("❌ Không kết nối được Local Server hoặc sai đường dẫn: " .. fullUrl)
    end
    return nil
end
