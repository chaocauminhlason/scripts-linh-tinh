-- ====================================================================
-- MODULE: AUTO FISCH (SOL'S RNG MINIGAME RADAR & AUTO CLICKER)
-- Tự động theo dõi thanh Minigame và thực hiện nhấp chuột an toàn
-- ====================================================================
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local LocalPlayer = Players.LocalPlayer

return function(Tabs, Fluent, Utils)
    getgenv().TargetTrackerOnly = false

    local cachedObj1 = nil
    local cachedObj2 = nil

    local function getMinigameBarsCached(mainUI)
        if cachedObj1 and cachedObj2 and cachedObj1.Parent and cachedObj2.Parent and cachedObj1.Visible and cachedObj2.Visible then
            return cachedObj1, cachedObj2
        end

        cachedObj1 = nil
        cachedObj2 = nil

        if not mainUI then return nil, nil end

        for _, desc in ipairs(mainUI:GetDescendants()) do
            if desc:IsA("ImageLabel") and desc.Visible then
                local children = desc:GetChildren()
                local foundTarget = nil
                local foundForce = nil

                for _, child in ipairs(children) do
                    if child:IsA("ImageLabel") then
                        if math.abs(child.Size.X.Scale - 0.16) < 0.02 then
                            foundTarget = child
                        elseif child.Position.X.Scale == 0 and math.abs(child.Size.Y.Scale - 0.2) < 0.05 then
                            foundForce = child
                        end
                    end
                end

                if foundTarget and foundForce then
                    cachedObj1 = foundTarget
                    cachedObj2 = foundForce
                    return cachedObj1, cachedObj2
                end
            end
        end

        return nil, nil
    end

    local function processTargetTracking()
        if not getgenv().TargetTrackerOnly then return end

        local playerGui = LocalPlayer:FindFirstChild("PlayerGui")
        local mainUI = playerGui and playerGui:FindFirstChild("MainInterface")
        if not mainUI then return end

        local obj1, obj2 = getMinigameBarsCached(mainUI)

        if obj1 and obj2 and obj1.Visible and obj2.Visible then
            local targetPos = obj1.Position.X.Scale
            local targetWidth = obj1.Size.X.Scale or 0.16
            local targetQuarter = targetPos + (targetWidth / 4)
            local forceLength = obj2.Size.X.Scale

            if forceLength < targetQuarter then
                if Utils and type(Utils.ThrottledClick) == "function" then
                    Utils.ThrottledClick(400, 300)
                end
            end
        end
    end

    --------------------------------------------------------------------------------
    -- TAB CONTROLS
    --------------------------------------------------------------------------------
    local AutoFischToggle = Tabs.Main:AddToggle("AutoFischToggle", { Title = "Auto Fisch", Default = false })

    AutoFischToggle:OnChanged(function()
        getgenv().TargetTrackerOnly = AutoFischToggle.Value
        
        if getgenv().TargetTrackerConnection1 then 
            getgenv().TargetTrackerConnection1:Disconnect() 
            getgenv().TargetTrackerConnection1 = nil
        end

        if AutoFischToggle.Value then
            getgenv().TargetTrackerConnection1 = RunService.RenderStepped:Connect(processTargetTracking)
            Fluent:Notify({ Title = "Auto Fisch", Content = "Started with 120ms safety click limit.", Duration = 2 })
        else
            Fluent:Notify({ Title = "Auto Fisch", Content = "Stopped completely.", Duration = 2 })
        end
    end)
end
