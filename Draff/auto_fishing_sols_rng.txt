-- =============================================================
-- AUTO FISHING SOL'S RNG (100% CATCH SUCCESS & AUTO WIN)
-- =============================================================

local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local RunService = game:GetService("RunService")
local VirtualInputManager = game:GetService("VirtualInputManager")
local LocalPlayer = Players.LocalPlayer

local Quad = require(ReplicatedStorage:WaitForChild("Modules"):WaitForChild("UI"):WaitForChild("Quad"))
local FishStore = Quad.Store.GetStore("FishUI")
local FishingPacket = require(ReplicatedStorage:WaitForChild("Packets"):WaitForChild("Fishing"))

getgenv().AutoFishing = true

-- Hủy vòng lặp cũ nếu đã chạy
if getgenv().AutoFishingConnection then
    getgenv().AutoFishingConnection:Disconnect()
end

print("[Auto-Fishing] Đã khởi chạy kịch bản Auto Fishing Sol's RNG...")

local lastClickTime = 0

-- Hàm nhấp đúp mượt mà
local function doubleClick(x, y)
    VirtualInputManager:SendMouseButtonEvent(x, y, 0, true, game, 0)
    task.wait(0.02)
    VirtualInputManager:SendMouseButtonEvent(x, y, 0, false, game, 0)
    task.wait(0.03)
    VirtualInputManager:SendMouseButtonEvent(x, y, 0, true, game, 0)
    task.wait(0.02)
    VirtualInputManager:SendMouseButtonEvent(x, y, 0, false, game, 0)
end

getgenv().AutoFishingConnection = RunService.Heartbeat:Connect(function()
    if not getgenv().AutoFishing then return end

    local now = tick()
    local mainUI = LocalPlayer.PlayerGui:FindFirstChild("MainInterface")

    -- 1. KHI MINIGAME KÍCH HOẠT: Mở rộng vùng mục tiêu 100% & Tự gửi tín hiệu thắng
    if FishStore.isMinigameStarted or FishStore.isMinigameMode then
        pcall(function()
            -- Mở rộng khối màu xanh an toàn trùm 100% thanh slider
            FishStore.successWidth = 1.0
            FishStore.catchValue = 0.5
            FishStore.successPosition = 0.5
            
            -- Đẩy tiến trình kéo cá mượt mà
            local p = FishStore.catchProgress or 0
            if p < 0.9 then
                FishStore.catchProgress = p + 0.08
            else
                -- Gửi tín hiệu hoàn thành minigame cho Server
                FishingPacket.gameResult.send(true)
            end
        end)

        -- Giữ click nhấp đúp
        doubleClick(400, 300)
        return
    end

    -- 2. KHI BẢNG KẾT QUẢ HIỂN THỊ: Đóng bảng qua nút X (MainInterface:GetChildren()[56]:GetChildren()[7].ImageButton)
    local popupOpen = false
    if mainUI then
        for _, desc in ipairs(mainUI:GetDescendants()) do
            if desc:IsA("TextLabel") and desc.Visible then
                if desc.Text:find("trốn thoát") or desc.Text:find("thất bại") or desc.Text:find("bắt được") then
                    popupOpen = true
                    break
                end
            end
        end
    end

    if popupOpen and now - lastClickTime > 0.4 then
        lastClickTime = now
        
        -- Đóng trực tiếp qua Nút X chính xác được dò ra: [56][7].ImageButton
        pcall(function()
            local xBtn = mainUI:GetChildren()[56]
                and mainUI:GetChildren()[56]:GetChildren()[7]
                and mainUI:GetChildren()[56]:GetChildren()[7]:FindFirstChild("ImageButton")
            
            if xBtn then
                local pos = xBtn.AbsolutePosition
                local size = xBtn.AbsoluteSize
                doubleClick(pos.X + size.X/2, pos.Y + size.Y/2 + 58)
                xBtn.Visible = false
            end
        end)

        FishStore.isResultMode = false
        print("[Auto-Fishing] Đã đóng cửa sổ thông báo kết quả qua nút X [56][7]!")
        return
    end

    -- 3. KHI NÚT "CÂU" SẴN SÀNG: Tự động thả cần
    local fishBtnReady = false
    if mainUI then
        for _, desc in ipairs(mainUI:GetDescendants()) do
            if (desc:IsA("TextLabel") or desc:IsA("TextButton")) and desc.Text == "Fish" then
                local btn = desc:IsA("TextButton") and desc or desc.Parent
                if btn and btn:IsA("GuiObject") and btn.Visible then
                    fishBtnReady = true
                    break
                end
            end
        end
    end

    if fishBtnReady and not FishStore.isFishing and now - lastClickTime > 1.2 then
        lastClickTime = now
        
        -- Thả câu trực tiếp qua Packet ByteNet
        local FishingPoints = workspace.Map.Miscs.FishingPoints
        local playerPos = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") and LocalPlayer.Character.HumanoidRootPart.Position
        
        if playerPos then
            for _, spot in ipairs(FishingPoints:GetChildren()) do
                if spot:IsA("Model") then
                    local p = spot:GetPivot().Position
                    if (p - playerPos).Magnitude < 15 then
                        local spotId = spot:GetAttribute("spotId")
                        if spotId then 
                            FishingPacket.start.send(spotId) 
                            print("[Auto-Fishing] Đã thả cần câu qua Packet!")
                        end
                        break
                    end
                end
            end
        end
    end
end)
